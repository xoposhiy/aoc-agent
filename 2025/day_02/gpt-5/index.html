    # Advent of Code 2025 Day 2 - Gift Shop
#
# Part 1: Sum all IDs within the given (merged) ranges that are a repeated pair of the same k-digit block.
# Numbers of the form V = S repeated twice, with S in [10^(k-1), 10^k - 1], are exactly
# V = S * (10^k + 1). For each k, the set is an arithmetic progression with factor F=10^k+1
# and S varying. We intersect with each interval and sum using arithmetic series formula.
# This avoids enumerating each S individually.
#
# Part 2: An ID is invalid if it is made only of some sequence repeated at least twice.
# Equivalently, there exists k>=1 and m>=2 and S (k-digit, no leading zero) such that
# V = S * sum_{j=0..m-1} 10^{j k}. The naive sum over (k,m) double-counts numbers that
# have non-primitive S (e.g., 111111 has S=11,m=3 and S=1,m=6). To count each invalid
# number exactly once, we count only primitive S of length k (S that are not themselves a
# repetition of a shorter block), across m>=2. Every periodic number has a unique minimal
# primitive k (its period), so this is a partition.
#
# For a fixed k, let proper divisors d of k be D. Define H_d = (10^k - 1) // (10^d - 1).
# A k-digit number S is non-primitive iff S is divisible by H_d for some proper divisor d.
# So primitive S are those not divisible by any H_d (d in D).
# We need the sum of S in an interval [s1, s2] that are not divisible by any H_d.
# We compute this with inclusion-exclusion over D: for any subset J subset D, let
#   M_J = lcm(H_d for d in J), sign = (-1)^{|J|}.
# Then sum_primitive(S in [s1,s2]) = sum_{J subset D} sign * sum_{S in [s1,s2], S multiple of M_J} S.
# The empty set J corresponds to M_empty=1 with sign +1 (sum of all S in [s1,s2]).
#
# For each (k,m): Factor F_km = sum_{j=0..m-1} 10^{j k}. Allowed S in [10^(k-1), 10^k-1]. The values V
# cover the AP [F*s_lo, F*s_hi]. For every merged interval [a,b], intersect with this span, convert
# to S-range via s1=ceil(L/F), s2=floor(R/F), then sum primitive S using the IE precomputed for k.
# Multiply by F and add to total for Part 2.
#
# Complexity: k <= floor(digits(max)/2). For typical inputs (~10 digits), k <= 5. Proper divisors per k are few,
# inclusion-exclusion has up to 2^{tau(k)-1} terms (small). Intervals count is modest. Runtime << 1s.
from __future__ import annotations
import sys
import math
from typing import List, Tuple, Dict

Interval = Tuple[int, int]


def read_intervals(path: str) -> List[Interval]:
    with open(path, 'r', encoding='utf-8') as f:
        data = f.read()
    s = ''.join(ch for ch in data if ch not in '\n\r\t ')
    parts = [p for p in s.split(',') if p]
    intervals: List[Interval] = []
    for p in parts:
        if '-' not in p:
            continue
        a_str, b_str = p.split('-', 1)
        a = int(a_str)
        b = int(b_str)
        if a > b:
            a, b = b, a
        intervals.append((a, b))
    intervals.sort()
    # merge
    merged: List[Interval] = []
    if not intervals:
        return merged
    cs, ce = intervals[0]
    for s2,e2 in intervals[1:]:
        if s2 <= ce + 1:
            if e2 > ce:
                ce = e2
        else:
            merged.append((cs, ce))
            cs, ce = s2, e2
    merged.append((cs, ce))
    return merged


def sum_AP_multiples_in_range(M: int, lo: int, hi: int) -> Tuple[int, int]:
    """Return (count, sum) of numbers in [lo, hi] that are multiples of M.
    Assumes M >= 1. Uses integer arithmetic.
    """
    if lo > hi:
        return 0, 0
    if M <= 0:
        raise ValueError('M must be positive')
    r1 = (lo + M - 1) // M
    r2 = hi // M
    if r1 > r2:
        return 0, 0
    n = r2 - r1 + 1
    # sum of terms: M*(r1 + r2)*n/2
    return n, M * (r1 + r2) * n // 2


def precompute_k_structures(k: int, pow10: List[int]) -> Tuple[List[int], List[int]]:
    """For given k, compute the list of moduli H_d for proper divisors d of k.
    Returns (divisors_d, H_list) where divisors_d is the list of proper divisors d, and H_list[i] corresponds.
    H_d = (10^k - 1) // (10^d - 1).
    """
    # proper divisors of k (exclude k itself), include d=1
    divs: List[int] = []
    for d in range(1, k):
        if k % d == 0:
            divs.append(d)
    H_list: List[int] = []
    t10k = pow10[k]
    for d in divs:
        H = (t10k - 1) // (pow10[d] - 1)
        H_list.append(H)
    return divs, H_list


def subset_lcms(H_list: List[int]) -> List[Tuple[int, int]]:
    """Return list of (M, sign) for all subsets of H_list (including empty subset with M=1, sign=+1).
    sign = (-1)^{|subset|}. M is lcm of selected H's (1 for empty).
    """
    from math import gcd
    res: List[Tuple[int,int]] = []
    def lcm(a: int, b: int) -> int:
        return a // gcd(a, b) * b
    # iterative build
    res = [(1, +1)]
    for H in H_list:
        new_entries: List[Tuple[int,int]] = []
        for M, sign in res:
            M2 = lcm(M, H)
            new_entries.append((M2, -sign))
        res += new_entries
    return res


def sum_primitive_s_in_range(k: int, s1: int, s2: int, lcms_signed: List[Tuple[int,int]]) -> int:
    """Using inclusion-exclusion with precomputed (M, sign) list for this k, compute sum of all s in [s1,s2]
    that are not divisible by any H_d (proper divisor d of k)."""
    total = 0
    for M, sign in lcms_signed:
        cnt, ssum = sum_AP_multiples_in_range(M, s1, s2)
        if cnt:
            total += sign * ssum
    return total


def part1_sum(intervals: List[Interval], pow10: List[int]) -> int:
    if not intervals:
        return 0
    max_upper = intervals[-1][1]
    # max k where minimal 2k-digit number (10^(2k-1)) <= max_upper
    kmax = 0
    while True:
        k_next = kmax + 1
        if pow10[2*k_next - 1] <= max_upper:
            kmax = k_next
        else:
            break
    total = 0
    for k in range(1, kmax + 1):
        ten_k = pow10[k]
        F = ten_k + 1
        s_lo = ten_k // 10
        s_hi = ten_k - 1
        Lk = F * s_lo
        Rk = F * s_hi
        for a, b in intervals:
            if b < Lk:
                continue
            if a > Rk:
                break
            L = max(a, Lk)
            R = min(b, Rk)
            if L > R:
                continue
            s1 = (L + F - 1) // F
            s2 = R // F
            if s1 < s_lo:
                s1 = s_lo
            if s2 > s_hi:
                s2 = s_hi
            if s1 > s2:
                continue
            n, ssum = sum_AP_multiples_in_range(1, s1, s2)  # here M=1 meaning sum all integers in range
            total += F * ssum
    return total


def part2_sum(intervals: List[Interval], pow10: List[int]) -> int:
    if not intervals:
        return 0
    max_upper = intervals[-1][1]
    # Lmax = max digits = floor(log10(max_upper)) + 1
    Lmax = len(str(max_upper))
    total = 0
    for k in range(1, Lmax // 2 + 1):
        ten_k = pow10[k]
        s_lo = ten_k // 10
        s_hi = ten_k - 1
        # Precompute inclusion-exclusion structures for this k
        divs, H_list = precompute_k_structures(k, pow10)
        lcms_signed = subset_lcms(H_list)  # includes empty subset (1,+1)
        # Iterate m >= 2 until minimal value exceeds max
        m = 2
        while True:
            # Factor for (k,m): F = sum_{j=0}^{m-1} 10^{j k}
            F = (pow10[m*k] - 1) // (ten_k - 1)
            Lk = F * s_lo
            if Lk > max_upper:
                break
            Rk = F * s_hi
            for a,b in intervals:
                if b < Lk:
                    continue
                if a > Rk:
                    break
                L = max(a, Lk)
                R = min(b, Rk)
                if L > R:
                    continue
                # translate to s-space
                s1 = (L + F - 1) // F
                s2 = R // F
                if s1 < s_lo:
                    s1 = s_lo
                if s2 > s_hi:
                    s2 = s_hi
                if s1 > s2:
                    continue
                ssum_prim = sum_primitive_s_in_range(k, s1, s2, lcms_signed)
                if ssum_prim:
                    total += F * ssum_prim
            m += 1
    return total

    p2 = part2_sum(intervals, pow10)
    print(f"Part 1: {p1}")
    print(f"Part 2: {p2}")

if __name__ == '__main__':
    main()
