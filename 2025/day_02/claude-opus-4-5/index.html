    """
Day 2: Gift Shop - Find invalid IDs (numbers made of repeating patterns)

Approach:
- Instead of iterating through all numbers in each range (could be billions),
  we generate all possible "doubled" numbers (pattern repeated twice)
- A doubled number has even length, and first half equals second half
- Generate all such numbers up to the maximum value in any range
- Check which ones fall within any of the given ranges
- Sum them up
"""

def generate_repeated_patterns(max_val):
    """Generate all numbers that are patterns repeated at least twice."""
    result = set()

    pattern_len = 1
    while 10**(pattern_len * 2 - 1) <= max_val:
        if pattern_len == 1:
            start, end = 1, 9
        else:
            start, end = 10**(pattern_len - 1), 10**pattern_len - 1

        for pattern in range(start, end + 1):
            pattern_str = str(pattern)
            reps = 2
            while True:
                num = int(pattern_str * reps)
                if num > max_val:
                    break
                result.add(num)
                reps += 1

        pattern_len += 1

    return result

def generate_doubled_numbers(max_val):
    """Generate all doubled numbers (Part 1)."""
    result = []
    pattern_len = 1
    while True:
        if pattern_len == 1:
            start, end = 1, 9
        else:
            start, end = 10**(pattern_len-1), 10**pattern_len - 1

        for pattern in range(start, end + 1):
            doubled = int(str(pattern) * 2)
            if doubled > max_val:
                return result
            result.append(doubled)

        pattern_len += 1
        if 10**(pattern_len-1) > max_val:
            return result
    return result

def solve(input_text, part2=False):
    input_text = input_text.strip().rstrip(',')
    ranges = []
    for part in input_text.split(','):
        lo, hi = map(int, part.strip().split('-'))
        ranges.append((lo, hi))

    max_val = max(hi for lo, hi in ranges)

    if part2:
        candidates = generate_repeated_patterns(max_val)
    else:
        candidates = generate_doubled_numbers(max_val)

    total = 0
    for num in candidates:
        for lo, hi in ranges:
            if lo <= num <= hi:
                total += num
                break

    return total

with open('input.txt', 'r') as f:
    input_text = f.read()

print(f"Part 1: {solve(input_text, part2=False)}")
print(f"Part 2: {solve(input_text, part2=True)}")
