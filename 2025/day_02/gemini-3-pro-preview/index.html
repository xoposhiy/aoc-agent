    try:
        with open('input.txt', 'r') as f:
            data = f.read().strip()
    except FileNotFoundError:
        print("input.txt not found")
        return

    ranges_str = data.split(',')
    ranges = []
    for r in ranges_str:
        try:
            start, end = map(int, r.split('-'))
            ranges.append((start, end))
        except ValueError:
            continue

    total_invalid_sum = 0
    count = 0

    # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤—Å–µ —á–∏—Å–ª–∞ –≤ –¥–∏–∞–ø–∞–∑–æ–Ω–∞—Ö

            L = len(s)

            # –ò—â–µ–º, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ s –ø–æ–≤—Ç–æ—Ä–µ–Ω–∏–µ–º –ø–æ–¥—Å—Ç—Ä–æ–∫–∏ k —Ä–∞–∑ (k >= 2)
            # –≠—Ç–æ –∑–Ω–∞—á–∏—Ç len(P) = L // k.

            # –ü–µ—Ä–µ–±–∏—Ä–∞–µ–º –≤–æ–∑–º–æ–∂–Ω—ã–µ –¥–ª–∏–Ω—ã –ø–∞—Ç—Ç–µ—Ä–Ω–∞ d
