"""
AoC 2025 Day 1: Secret Entrance

Part 1: Count how many rotations end with the dial pointing at 0.
Part 2: Count ALL clicks that land on 0 (during rotation + at end).

The dial has values 0-99 in a circle. Start at 50.
L<n> = rotate left (subtract), R<n> = rotate right (add).

For Part 2, we count zeros mathematically:
- Right from p by d: count multiples of 100 in range [p+1, p+d] = (p+d)//100
- Left from p by d: count non-positive multiples of 100 in [p-d, p-1]
  = floor((p-1)/100) - floor((p-d-1)/100)
"""

import math

def count_zeros_right(p, d):
    """Count clicks landing on 0 when rotating RIGHT from p by d."""
    return (p + d) // 100

def count_zeros_left(p, d):
    """Count clicks landing on 0 when rotating LEFT from p by d."""
    return math.floor((p - 1) / 100) - math.floor((p - d - 1) / 100)

def solve():
    with open("input.txt") as f:
        rotations = [line.strip() for line in f if line.strip()]
    
    position = 50
    zero_count_p1 = 0
    zero_count_p2 = 0
    
    for rotation in rotations:
        direction = rotation[0]
        distance = int(rotation[1:])
        
        if direction == 'L':
            zeros = count_zeros_left(position, distance)
            position = (position - distance) % 100
        else:  # R
            zeros = count_zeros_right(position, distance)
            position = (position + distance) % 100
        
        zero_count_p2 += zeros
        if position == 0:
            zero_count_p1 += 1
    
    print(f"Part 1: {zero_count_p1}")
    print(f"Part 2: {zero_count_p2}")

solve()
