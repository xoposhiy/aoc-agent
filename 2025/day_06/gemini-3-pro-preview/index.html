    with open('input.txt', 'r') as f:
        lines = f.read().splitlines()

    cols = max_len

    # Identify empty columns
    empty_cols = []
    for c in range(cols):
        is_empty = True
        for r in range(rows):
            if grid[r][c] != ' ':
                is_empty = False
                break
        if is_empty:
            empty_cols.append(c)

    # Find ranges of non-empty columns
    # We can treat empty columns as splitters
    # Or simply iterate and group
    problem_blocks = []
    start = -1
    for c in range(cols):
        is_empty = True
        for r in range(rows):
            if grid[r][c] != ' ':
                is_empty = False
                break

        if not is_empty:
            if start == -1:
                start = c
        else:
            if start != -1:
                problem_blocks.append((start, c))
                start = -1
    if start != -1:
        problem_blocks.append((start, cols))

    grand_total = 0

    for c_start, c_end in problem_blocks:
        # Extract block
        numbers = []
        operator = None

        for r in range(rows):
            segment = grid[r][c_start:c_end].strip()

                continue

                operator = segment
            else:
                try:
                    numbers.append(int(segment))

                    # Could happen if parsing is wrong, but shouldn't based on problem

            # Should not happen
            print(f"No operator found in block {c_start}-{c_end}")
            continue

        if not numbers:
             # Should not happen
            print(f"No numbers found in block {c_start}-{c_end}")
            continue

        result = numbers[0]
        if operator == '+':
            for n in numbers[1:]:
                result += n
        elif operator == '*':
            for n in numbers[1:]:
                result *= n

        grand_total += result

    import sys

def solve():
    """
    Solves AoC 2025 Day 6 Parts 1 and 2.
    """
    try:
        with open('input.txt', 'r') as f:
            lines = f.read().splitlines()
    except FileNotFoundError:
        print("input.txt not found.")
        return

    if not lines:
        return

    # Normalize grid dimensions
    max_len = max(len(l) for l in lines)
    grid = [l.ljust(max_len) for l in lines]
    rows = len(grid)
    cols = max_len

    # Identify problem blocks (separated by empty columns)
    problem_blocks = []
    start = -1
    for c in range(cols):
        is_empty = True
        for r in range(rows):
            if grid[r][c] != ' ':
                is_empty = False
                break

        if not is_empty:
            if start == -1:
                start = c
        else:
            if start != -1:
                problem_blocks.append((start, c))
                start = -1
    if start != -1:
        problem_blocks.append((start, cols))

    total_p1 = 0
    total_p2 = 0

    for c_start, c_end in problem_blocks:
        # Extract block content
        block_cols = [
            "".join(grid[r][c] for r in range(rows))
            for c in range(c_start, c_end)
        ]

        # Find operator
        operator = None
        for col_str in block_cols:
            for char in col_str:
                if char in ('+', '*'):
                    operator = char
                    break
            if operator:
                break

        if not operator:
            continue

        # Part 1: Horizontal numbers
        # Each row (stripped) is a number or the operator
        nums_p1 = []
        for r in range(rows):
            row_segment = grid[r][c_start:c_end].strip()
            if not row_segment:
                continue
            # It's either a number or the operator line
            # The operator line might contain just the operator
            if row_segment in ('+', '*'):
                continue

            # If the row has digits, parse it
            # Note: The operator might be on the same line as numbers in some weird layout?
            # Task says "at the bottom", and example shows it on its own line effectively 
            # or at least aligned. 
            # But wait, my Part 1 code handled "row segment is '+' or '*'" as operator check.
            # And parsed numbers otherwise.
            # Let's stick to that logic which worked.
            if row_segment.isdigit():
                 nums_p1.append(int(row_segment))

        # Calculate Part 1
        if nums_p1:
            res = nums_p1[0]
            for x in nums_p1[1:]:
                if operator == '+': res += x
                else: res *= x
            total_p1 += res

        # Part 2: Vertical numbers
        # Each column (digits only) is a number
        nums_p2 = []
        for col_str in block_cols:
            digits = [ch for ch in col_str if ch.isdigit()]
            if digits:
                nums_p2.append(int("".join(digits)))

        # Calculate Part 2
        if nums_p2:
            res = nums_p2[0]
            for x in nums_p2[1:]:
                if operator == '+': res += x
                else: res *= x
            total_p2 += res

    print(f"Part 1: {total_p1}")
    print(f"Part 2: {total_p2}")

        if not is_empty:
            if start == -1: start = c
        else:
            if start != -1:
                blocks.append((start, c))
                start = -1
    if start != -1:
        blocks.append((start, vis_cols))

    # SVG Params
    char_w = 12
    char_h = 20
    margin_x = 20
    margin_y = 40
    header_h = 30

    width = margin_x * 2 + vis_cols * char_w
    height = margin_y * 3 + rows * char_h * 2 + header_h * 2

    svg = [f'<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}" style="background-color:#1e1e1e; font-family: monospace; font-size: 14px;">']

    colors = ['#FF6B6B', '#4ECDC4', '#FFE66D', '#1A535C', '#F7FFF7']
    op_color = '#FF9F1C'

    def draw_grid(offset_y, title, mode):
        svg.append(f'<text x="{margin_x}" y="{offset_y - 10}" fill="#ffffff" font-weight="bold">{title}</text>')

        # Draw characters and backgrounds
        for c_start, c_end in blocks:
            # Find operator
            operator = None
            for c in range(c_start, c_end):
                for r in range(rows):
                    if grid_vis[r][c] in ('+', '*'):
                        operator = grid_vis[r][c]

            # Determine number regions
            regions = [] # list of (r, c, color_index)

            if mode == 'p1':
                # Horizontal numbers
                color_idx = 0
                for r in range(rows):
                    segment = grid_vis[r][c_start:c_end].strip()
                    if not segment: continue
                    if segment in ('+', '*'):
                         # Operator pos
                         col_idx_in_block = grid_vis[r][c_start:c_end].find(segment)
                         regions.append([(r, c_start + col_idx_in_block, -1)])
                         continue

                    if segment.isdigit():
                        # Find start/end col
                        # Note: numbers might be aligned differently
                        # Simple approach: highlight all non-space in this row/block
                        cells = []
                        for c in range(c_start, c_end):
                            if grid_vis[r][c] != ' ':
                                cells.append((r, c, color_idx))
                        regions.append(cells)
                        color_idx += 1
            else:
                # Vertical numbers
                color_idx = 0
                for c in range(c_start, c_end):
                    cells = []
                    is_op_col = False
                    for r in range(rows):
                         if grid_vis[r][c] in ('+', '*'):
                             cells.append((r, c, -1))
                             is_op_col = True
                         elif grid_vis[r][c].isdigit():
                             cells.append((r, c, color_idx))

                    if cells:
                        # If the column only contained operator, it's op color (handled by -1)
                        # If it contained numbers, it's a number
                        # But wait, operator is separate from number in P2?
                        # In P2, column forms a number. Operator is separate.
                        # But operator char is in the grid.
                        regions.append(cells)
                        if not all(x[2] == -1 for x in cells):
                             color_idx += 1

            # Draw backgrounds
            for cells in regions:
                if not cells: continue
                # Pick color
                c_code = colors[cells[0][2] % len(colors)] if cells[0][2] != -1 else op_color

                for r, c, _ in cells:
                    x = margin_x + c * char_w
                    y = offset_y + r * char_h
                    svg.append(f'<rect x="{x}" y="{y}" width="{char_w}" height="{char_h}" fill="{c_code}" opacity="0.4" />')

        # Draw text on top
        for r in range(rows):
            for c in range(vis_cols):
                char = grid_vis[r][c]
                x = margin_x + c * char_w + char_w/2
                y = offset_y + r * char_h + char_h/2 + 5
                fill = "#ffffff"
                if char == ' ': continue
                svg.append(f'<text x="{x}" y="{y}" text-anchor="middle" fill="{fill}">{char}</text>')

    draw_grid(margin_y + header_h, "Part 1: Horizontal Parsing (Rows)", 'p1')
    draw_grid(margin_y * 2 + rows * char_h + header_h * 2, "Part 2: Vertical Parsing (Columns)", 'p2')

    svg.append('</svg>')

    with open('visualization.svg', 'w') as f:
        f.write("".join(svg))

    print("SVG generated.")

if __name__ == '__main__':
