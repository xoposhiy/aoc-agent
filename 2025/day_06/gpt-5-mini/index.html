    # Approach:
# - Read the input as lines, pad to equal width.
# - A separator column is a column where every character (all rows) is a space.
# - Split the grid into contiguous column segments separated by such separator columns.
# - For each segment, the last row contains the operation character (+ or *).
# - For each row above the last, take the substring covering the segment and strip spaces; if non-empty, parse as integer.
# - Compute the result for the segment: sum or product of the collected integers according to the operation.
# - Sum all segment results and print the grand total.

from math import prod

def solve_part1(lines):
    H = len(lines)
    W = max(len(l) for l in lines)
    # pad lines
    grid = [l.ljust(W) for l in lines]
    # identify separator columns
    sep = [all(grid[r][c] == ' ' for r in range(H)) for c in range(W)]
    parts = []
    c = 0
    while c < W:
        if sep[c]:
            c += 1
            continue
        start = c
        while c < W and not sep[c]:
            c += 1
        end = c  # segment covers [start, end)
        parts.append((start, end))
    total = 0
    for start, end in parts:
        op_char = None
        # operation is in last row
        for ch in grid[-1][start:end]:
            if ch != ' ':
                op_char = ch
                break
        if op_char is None:
            continue
        nums = []
        for r in range(H-1):
            s = grid[r][start:end].strip()
            if s:
                # s may contain multiple numbers separated by spaces if formatting odd,
                # but within a problem vertical numbers shouldn't be separated horizontally.
                # However split by whitespace and take each token as a number.
                for tok in s.split():
                    nums.append(int(tok))
        if not nums:
            continue
        if op_char == '+':
            val = sum(nums)
        elif op_char == '*':
            val = prod(nums)
        else:
            raise ValueError('Unknown op: %r' % op_char)
        total += val
    return total

if __name__ == '__main__':

    ans1 = solve_part1(lines)
    print(f"Part 1: {ans1}")
