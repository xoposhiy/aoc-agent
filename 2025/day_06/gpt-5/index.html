    # AoC 2025 Day 6 - Part 1
# Idea:
# The input is a horizontal strip with multiple problems separated by at least one full column of spaces.
# Each problem occupies a contiguous block of columns. Within that block:
#  - The bottom row contains the operator ('+' or '*') somewhere inside the block.
#  - Every other row that contains non-space characters inside the block represents exactly one number.
# We parse by columns:
# 1) Pad all lines to equal width. Compute for each column whether it's entirely spaces.
# 2) Split the width into segments of consecutive non-empty columns. Each segment is one problem.
# 3) For each segment, scan rows 0..H-2: take substring [l:r], strip, if non-empty, parse as int and collect.
# 4) Bottom row H-1 in [l:r] should contain '+' or '*'; determine op.
# 5) Compute sum or product of the collected numbers accordingly.
# 6) Grand total is sum of results over all segments.
# Assumptions: Numbers are non-negative and contain only digits; lines may contain trailing spaces that are significant.
# Complexity: O(H*W), where H is number of lines and W is the width.

from pathlib import Path
import sys

p = Path('input.txt')
raw = p.read_text().splitlines()

# Normalize width
W = max((len(line) for line in raw), default=0)
H = len(raw)
lines = [line.ljust(W) for line in raw]

# Identify empty columns
col_empty = [True]*W
for j in range(W):
    for i in range(H):
        if lines[i][j] != ' ':
            col_empty[j] = False
            break

# Extract segments [l, r) of consecutive non-empty columns
segments = []
j = 0
while j < W:
    while j < W and col_empty[j]:
        j += 1
    if j >= W:
        break
    l = j
    while j < W and not col_empty[j]:
        j += 1
    r = j  # exclusive
    segments.append((l, r))

# Helper: safe strip and check digits

def parse_row_number(s: str):
    s = s.strip()
    if not s:
        return None
    if not s.isdigit():
        # Defensive: ignore any non-digit junk inside problem regions except bottom operator row.
        # However, according to the problem, it should be only digits/spaces.
        raise ValueError(f"Unexpected non-digit content in number row: {s!r}")
    return int(s)


def compute_problem(l: int, r: int) -> int:
    # Find operator in bottom row within [l, r)
    bottom = lines[-1][l:r]
    op = None
    if '+' in bottom:
        op = '+'
    if '*' in bottom:
        # If both present (shouldn't happen), it's ambiguous; but pick the one that actually appears as a single char.
        # Prefer unique; otherwise, raise.
        if op is None:
            op = '*'
        else:
            # Both found: ambiguous input
            raise ValueError("Both '+' and '*' found in the bottom row of a single problem block")
    if op is None:
        raise ValueError("No operator found in problem block")

    nums = []
    for i in range(H-1):  # all rows except bottom
        chunk = lines[i][l:r]
        s = chunk.strip()
        if not s:
            continue
        # Allow that row may contain operator-like symbols by accident? Spec says numbers only.
        if not s.isdigit():
            raise ValueError(f"Row {i} in block [{l},{r}) contains non-digit: {s!r}")
        nums.append(int(s))

    if not nums:
        return 0

    if op == '+':
        return sum(nums)
    else:
        prod = 1
        for v in nums:
            prod *= v
        return prod


grand_total = 0
for l, r in segments:
    grand_total += compute_problem(l, r)

print(f"Part 1: {grand_total}")

    # AoC 2025 Day 6 - Part 2
# Now numbers are given as columns, read right-to-left, one column per number within each problem block.
# Details per problem block [l, r):
#  - Bottom row contains operator ('+' or '*').
#  - Within the block, scan column indices from r-1 down to l.
#  - For each column c, compose a number by reading digits from top (row 0) to row H-2 (since H-1 is operator row), concatenating digits where present.
#    Spaces are ignored; if the entire column above is spaces, that column doesn't form a number? The statement says each number is given in its own column, implying columns that contain any digit constitute a number; empty columns should not occur except separators, which are a full column of spaces (but separators lie between problems, not inside a block).
#    To be robust, we will skip columns that contain no digits.
#  - Evaluate op over the list of numbers (in any order since + and * are commutative); but we'll preserve right-to-left just in case.
#  - Sum results across problems.

from pathlib import Path

raw = Path('input.txt').read_text().splitlines()
W = max((len(line) for line in raw), default=0)
H = len(raw)
lines = [line.ljust(W) for line in raw]

# Identify empty columns (entirely spaces)
col_empty = [True]*W
for j in range(W):
    for i in range(H):
        if lines[i][j] != ' ':
            col_empty[j] = False
            break

# Segments of consecutive non-empty columns are problem blocks
segments = []
j = 0
while j < W:
    while j < W and col_empty[j]:
        j += 1
    if j >= W:
        break
    l = j
    while j < W and not col_empty[j]:
        j += 1
    r = j
    segments.append((l, r))

# Helper to build number from a single column c (using rows 0..H-2)

def number_from_column(c: int) -> int | None:
    digits = []
    for i in range(H-1):
        ch = lines[i][c]
        if ch == ' ':
            continue
        if not ch.isdigit():
            # Unexpected symbol - be strict
            raise ValueError(f"Non-digit in digit column: {ch!r} at row {i}, col {c}")
        digits.append(ch)
    if not digits:
        return None
    return int(''.join(digits))


def compute_problem(l: int, r: int) -> int:
    bottom = lines[-1][l:r]
    op = None
    if '+' in bottom:
        op = '+'
    if '*' in bottom:
        if op is None:
            op = '*'
        else:
            raise ValueError("Both operators found in a block")
    if op is None:
        raise ValueError("No operator in a block")

    nums = []
    for c in range(r-1, l-1, -1):
        val = number_from_column(c)
        if val is None:
            continue
        nums.append(val)

    if not nums:
        return 0

    if op == '+':
        return sum(nums)
    prod = 1
    for v in nums:
        prod *= v
    return prod


grand_total = 0
for l, r in segments:
    grand_total += compute_problem(l, r)

print(f"Part 2: {grand_total}")

    # AoC 2025 Day 6 - Trash Compactor
# Approach
# We parse the worksheet as a grid of characters. Problems are contiguous blocks of columns
# separated by fully-empty (all-space) columns. The bottom row of each block contains an operator.
# Part 1: numbers are given as horizontal rows within a block (any non-empty row except bottom).
# Part 2: numbers are given as vertical columns within a block, read right-to-left; digits are
# gathered from top to the row above bottom, ignoring spaces.
# The implementation shares common parsing (grid, block segmentation) and computes both parts.

from __future__ import annotations
from pathlib import Path
from typing import List, Tuple


def read_grid(path: str = 'input.txt') -> list[str]:
    lines = Path(path).read_text().splitlines()
    if not lines:
        return []
    W = max(len(s) for s in lines)
    return [s.ljust(W) for s in lines]


def empty_columns_mask(grid: List[str]) -> list[bool]:
    if not grid:
        return []
    H, W = len(grid), len(grid[0])
    mask = [True] * W
    for j in range(W):
        for i in range(H):
            if grid[i][j] != ' ':
                mask[j] = False
                break
    return mask


def blocks_from_mask(mask: List[bool]) -> List[Tuple[int, int]]:
    W = len(mask)
    res = []
    j = 0
    while j < W:
        while j < W and mask[j]:
            j += 1
        if j >= W:
            break
        l = j
        while j < W and not mask[j]:
            j += 1
        res.append((l, j))
    return res


def op_in_block(grid: List[str], l: int, r: int) -> str:
    bottom = grid[-1][l:r]
    plus = '+' in bottom
    star = '*' in bottom
    if plus and star:
        raise ValueError('Ambiguous operator in block')
    if plus:
        return '+'
    if star:
        return '*'
    raise ValueError('No operator in block')


def solve_part1(grid: List[str], blocks: List[Tuple[int, int]]) -> int:
    H = len(grid)
    total = 0
    for l, r in blocks:
        op = op_in_block(grid, l, r)
        nums = []
        for i in range(H - 1):
            s = grid[i][l:r].strip()
            if not s:
                continue
            if not s.isdigit():
                raise ValueError(f'Non-digit row content: {s!r}')
            nums.append(int(s))
        if op == '+':
            total += sum(nums)
        else:
            prod = 1
            for v in nums:
                prod *= v
            total += prod
    return total


def number_from_column(grid: List[str], c: int) -> int | None:
    H = len(grid)
    digits = []
    for i in range(H - 1):
        ch = grid[i][c]
        if ch == ' ':
            continue
        if not ch.isdigit():
            raise ValueError(f'Unexpected char in number column: {ch!r}')
        digits.append(ch)
    if not digits:
        return None
    return int(''.join(digits))


def solve_part2(grid: List[str], blocks: List[Tuple[int, int]]) -> int:
    total = 0
    for l, r in blocks:
        op = op_in_block(grid, l, r)
        nums = []
        for c in range(r - 1, l - 1, -1):
            v = number_from_column(grid, c)
            if v is not None:
                nums.append(v)
        if op == '+':
            total += sum(nums)
        else:
            prod = 1
            for v in nums:
                prod *= v
            total += prod
    return total


def main():
    grid = read_grid('input.txt')
    mask = empty_columns_mask(grid)
    blocks = blocks_from_mask(mask)
    ans1 = solve_part1(grid, blocks)
    ans2 = solve_part2(grid, blocks)
    print(f'Part 1: {ans1}')
    print(f'Part 2: {ans2}')


if __name__ == '__main__':
    main()
