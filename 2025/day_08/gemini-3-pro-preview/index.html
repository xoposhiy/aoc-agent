    import math
import collections

def solve():
    with open('input.txt', 'r') as f:
        lines = f.readlines()

    points = []
    for line in lines:
        parts = line.strip().split(',')
        if len(parts) == 3:
            points.append(tuple(map(int, parts)))

    n = len(points)

    # Calculate all pairwise distances
    edges = []
    for i in range(n):
        for j in range(i + 1, n):
            p1 = points[i]
            p2 = points[j]
            dist_sq = (p1[0]-p2[0])**2 + (p1[1]-p2[1])**2 + (p1[2]-p2[2])**2
            edges.append((dist_sq, i, j))

    # Sort by distance (squared)
    edges.sort(key=lambda x: x[0])

    # --- Part 1 ---
    parent = list(range(n))
    def find(i):
        if parent[i] != i:
            parent[i] = find(parent[i])
        return parent[i]

    def union(i, j):
        root_i = find(i)
        root_j = find(j)
        if root_i != root_j:
            parent[root_i] = root_j
            return True
        return False

    limit = 1000
    for k in range(limit):
        _, u, v = edges[k]
        union(u, v)

    counts = collections.defaultdict(int)
    for i in range(n):
        counts[find(i)] += 1

    sizes = list(counts.values())
    sizes.sort(reverse=True)

    ans1 = 0
    if len(sizes) >= 3:
        ans1 = sizes[0] * sizes[1] * sizes[2]

    print(f"Part 1 Answer: {ans1}")

    # --- Part 2 ---
    # Reset DSU
    parent = list(range(n))
    num_components = n
    ans2 = 0

    # Iterate through ALL edges this time (or until connected)
    for dist_sq, u, v in edges:
        if union(u, v):
            num_components -= 1
            if num_components == 1:
                # This is the last connection needed
                ans2 = points[u][0] * points[v][0]
                break

    print(f"Part 2 Answer: {ans2}")

    # For visualization, we can just dump the points to a file or print some stats
    # I'll create a viz script separately.

if __name__ == '__main__':
    solve()

    n = len(points)

    # Calculate MST edges
    edges = []
    for i in range(n):
        for j in range(i + 1, n):
            p1 = points[i]
            p2 = points[j]
            dist_sq = (p1[0]-p2[0])**2 + (p1[1]-p2[1])**2 + (p1[2]-p2[2])**2
            edges.append((dist_sq, i, j))

    parent = list(range(n))

        if parent[i] != i:

        return parent[i]

    def union(i, j):
