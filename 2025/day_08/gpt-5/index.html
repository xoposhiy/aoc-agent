import os
from collections import defaultdict
import heapq

class KDNode:
    __slots__ = ("idxs", "bbox", "left", "right", "axis", "id")
    _id_counter = 0
    def __init__(self, idxs, bbox, left=None, right=None, axis=0):
        self.idxs = idxs  # list[int] if leaf else None to save memory
        self.bbox = bbox  # (minx,miny,minz,maxx,maxy,maxz)
        self.left = left
        self.right = right
        self.axis = axis
        self.id = KDNode._id_counter
        KDNode._id_counter += 1


def build_kdtree(points, leaf_size=16, axis=0, idxs=None):
    # points: list of (x,y,z)
    n = len(points)
    if idxs is None:
        idxs = list(range(n))
    # compute bbox
    minx = min(points[i][0] for i in idxs)
    miny = min(points[i][1] for i in idxs)
    minz = min(points[i][2] for i in idxs)
    maxx = max(points[i][0] for i in idxs)
    maxy = max(points[i][1] for i in idxs)
    maxz = max(points[i][2] for i in idxs)
    bbox = (minx,miny,minz,maxx,maxy,maxz)
    if len(idxs) <= leaf_size:
        return KDNode(list(idxs), bbox, left=None, right=None, axis=axis)
    # choose axis by spread
    spreads = [maxx-minx, maxy-miny, maxz-minz]
    axis = max(range(3), key=lambda a: spreads[a])
    # sort idxs by coordinate axis and split median
    idxs.sort(key=lambda i: points[i][axis])
    mid = len(idxs)//2
    left_idxs = idxs[:mid]
    right_idxs = idxs[mid:]
    left = build_kdtree(points, leaf_size=leaf_size, axis=(axis+1)%3, idxs=left_idxs)
    right = build_kdtree(points, leaf_size=leaf_size, axis=(axis+1)%3, idxs=right_idxs)
    return KDNode(None, bbox, left=left, right=right, axis=axis)


def box_min_dist2(b1, b2):
    # min squared distance between two axis-aligned boxes b1 and b2
    dx = 0.0
    for a in range(3):
        a_min1 = b1[a]
        a_max1 = b1[a+3]
        a_min2 = b2[a]
        a_max2 = b2[a+3]
        if a_max1 < a_min2:
            d = a_min2 - a_max1
        elif a_max2 < a_min1:
            d = a_min1 - a_max2
        else:
            d = 0.0
        dx += d*d
    return dx


def dist2(p, q):
    return (p[0]-q[0])**2 + (p[1]-q[1])**2 + (p[2]-q[2])**2


def k_closest_pairs(points, K, leaf_size=16):
    # Use best-first search over KD-tree node pairs to generate K closest point pairs by distance.
    # Returns list of tuples (d2, i, j) sorted by d2 ascending.
    n = len(points)
    if n < 2:
        return []
    root = build_kdtree(points, leaf_size=leaf_size)
    # priority queue: items are (priority_value, kind, counter, payload)
    # kind: 0 = nodepair, 1 = pointpair. Lower kind ensures node expansions before equal priority point pairs.
    pq = []
    counter = 0
    # Start with (root, root)
    heapq.heappush(pq, (0.0, 0, counter, (root, root)))
    counter += 1
    visited_node_pairs = set()
    out = []
    seen_point_pairs = set()

    def push_nodepair(a, b):
        aid = a.id
        bid = b.id
        key = (aid, bid) if aid <= bid else (bid, aid)
        if key in visited_node_pairs:
            return
        visited_node_pairs.add(key)
        lb = box_min_dist2(a.bbox, b.bbox)
        heapq.heappush(pq, (lb, 0, next_counter(), (a, b)))

    def next_counter():
        nonlocal counter
        c = counter
        counter += 1
        return c

    while pq and len(out) < K:
        priority, kind, _, payload = heapq.heappop(pq)
        if kind == 1:
            d2v, i, j = payload
            key = (i, j) if i < j else (j, i)
            if key in seen_point_pairs:
                continue
            seen_point_pairs.add(key)
            out.append((d2v, key[0], key[1]))
            continue
        # node pair
        a, b = payload
        # If both leaves, enumerate point pairs
        if a.idxs is not None and b.idxs is not None:
            if a is b:
                idxs_a = a.idxs
                la = len(idxs_a)
                for ii in range(la):
                    i = idxs_a[ii]
                    pi = points[i]
                    for jj in range(ii+1, la):
                        j = idxs_a[jj]
                        d2v = dist2(pi, points[j])
                        heapq.heappush(pq, (d2v, 1, next_counter(), (d2v, i, j)))
            else:
                idxs_a = a.idxs
                idxs_b = b.idxs
                # Enumerate cross pairs
                for i in idxs_a:
                    pi = points[i]
                    for j in idxs_b:
                        d2v = dist2(pi, points[j])
                        # ensure i<j consistently to avoid duplicates across symmetric node pairs
                        ii, jj = (i, j) if i < j else (j, i)
                        heapq.heappush(pq, (d2v, 1, next_counter(), (d2v, ii, jj)))
            continue
        # Not both leaves: expand the larger node
        # Choose which node to split: the one with larger bbox diagonal
        def box_diag2(node):
            b = node.bbox
            return (b[3]-b[0])**2 + (b[4]-b[1])**2 + (b[5]-b[2])**2
        # handle cases where one is leaf and other internal
        if a.idxs is None and b.idxs is None:
            # split the larger diagonal node
            if box_diag2(a) >= box_diag2(b):
                # expand a
                push_nodepair(a.left, b)
                push_nodepair(a.right, b)
            else:
                push_nodepair(a, b.left)
                push_nodepair(a, b.right)
        elif a.idxs is None:
            push_nodepair(a.left, b)
            push_nodepair(a.right, b)
        elif b.idxs is None:
            push_nodepair(a, b.left)
            push_nodepair(a, b.right)
        else:
            # Shouldn't happen (both leaves handled above)
            pass
    # Ensure sorted by distance
    out.sort(key=lambda t: (t[0], t[1], t[2]))
    return out[:K]


class DSU:
    __slots__ = ("p", "sz", "components")
    def __init__(self, n):
        self.p = list(range(n))
        self.sz = [1]*n
        self.components = n
    def find(self, x):
        p = self.p
        while p[x] != x:
            p[x] = p[p[x]]
            x = p[x]
        return x
    def union(self, a, b):
        ra = self.find(a); rb = self.find(b)
        if ra == rb:
            return False
        if self.sz[ra] < self.sz[rb]:
            ra, rb = rb, ra
        self.p[rb] = ra
        self.sz[ra] += self.sz[rb]
        self.components -= 1
        return True
    def sizes(self):
        # compute sizes of roots
        res = defaultdict(int)
        for i in range(len(self.p)):
            r = self.find(i)
            res[r] += 1
        return list(res.values())


def parse_input(text):
    pts = []
    for line in text.strip().splitlines():
        line = line.strip()
        if not line:
            continue
        x,y,z = map(int, line.split(','))
        pts.append((x,y,z))
    return pts


def solve_part1(points, K_pairs=1000):
    # get K closest pairs
    pairs = k_closest_pairs(points, K_pairs, leaf_size=16)
    dsu = DSU(len(points))
    for _, i, j in pairs:
        dsu.union(i, j)
    sizes = dsu.sizes()
    sizes.sort(reverse=True)
    while len(sizes) < 3:
        sizes.append(1)
    return sizes[0]*sizes[1]*sizes[2], pairs


def solve_part2(points):
    n = len(points)
    # We'll keep extracting closest pairs until components == 1.
    # But we cannot precompute all pairs (n^2). We'll stream from k_closest_pairs with increasing K.
    # Strategy: grow K exponentially until connectivity is achieved, then binary-search the exact edge.
    dsu = DSU(n)

    # helper: process first K pairs and return if connected and the edge where it happens
    def process_pairs(pairs, dsu_local):
        for d2v, i, j in pairs:
            dsu_local.union(i, j)
            if dsu_local.components == 1:
                return (i, j)
        return None

    # Exponential growth of K until connectivity achieved
    K = max(1000, n)  # start reasonably
    pairs = k_closest_pairs(points, K)
    dsu_tmp = DSU(n)
    found = process_pairs(pairs, dsu_tmp)
    while found is None:
        K *= 2
        pairs = k_closest_pairs(points, K)
        dsu_tmp = DSU(n)
        found = process_pairs(pairs, dsu_tmp)
        if K > n*50 and len(pairs) < K:
            # safety: all pairs exhausted? shouldn't happen
            break
    # Now we know that the last needed edge is within first K pairs.
    # Let's find the exact last union edge using a fresh DSU and scan.
    dsu2 = DSU(n)
    last_edge = None
    for _, i, j in pairs:
        if dsu2.union(i, j):
            last_edge = (i, j)
            if dsu2.components == 1:
                break
    if last_edge is None:
        # Fallback: compute more pairs
        more = k_closest_pairs(points, K*2)
        dsu2 = DSU(n)
        for _, i, j in more:
            if dsu2.union(i, j):
                last_edge = (i, j)
                if dsu2.components == 1:
                    pairs = more
                    break
    if last_edge is None:
        raise RuntimeError("Failed to connect graph with streamed pairs")
    a, b = last_edge
    return points[a][0] * points[b][0], pairs, last_edge

        if self.sz[ra] < self.sz[rb]:
            ra, rb = rb, ra
        self.p[rb] = ra
        self.sz[ra] += self.sz[rb]
        self.components -= 1
        return True
    def sizes(self):
        # compute sizes of roots
        res = defaultdict(int)
        for i in range(len(self.p)):
            r = self.find(i)
            res[r] += 1
        return list(res.values())


def parse_input(text):
    pts = []
    for line in text.strip().splitlines():
        line = line.strip()
        if not line:
            continue
        x,y,z = map(int, line.split(','))
        pts.append((x,y,z))
    return pts


def solve_part1(points, K_pairs=1000):
    # get K closest pairs
    pairs = k_closest_pairs(points, K_pairs, leaf_size=16)
    dsu = DSU(len(points))
    for _, i, j in pairs:
        dsu.union(i, j)
    sizes = dsu.sizes()
    sizes.sort(reverse=True)
    while len(sizes) < 3:
        sizes.append(1)
    return sizes[0]*sizes[1]*sizes[2], pairs


def solve_part2(points):
    n = len(points)
    # We'll keep extracting closest pairs until components == 1.
    # But we cannot precompute all pairs (n^2). We'll stream from k_closest_pairs with increasing K.
    # Strategy: grow K exponentially until connectivity is achieved, then binary-search the exact edge.
    dsu = DSU(n)

    # helper: process first K pairs and return if connected and the edge where it happens
    def process_pairs(pairs, dsu_local):
        for d2v, i, j in pairs:
            dsu_local.union(i, j)
            if dsu_local.components == 1:
                return (i, j)
        return None

    # Exponential growth of K until connectivity achieved
    K = max(1000, n)  # start reasonably
    pairs = k_closest_pairs(points, K)
    dsu_tmp = DSU(n)
    found = process_pairs(pairs, dsu_tmp)
    while found is None:
        K *= 2
        pairs = k_closest_pairs(points, K)
        dsu_tmp = DSU(n)
        found = process_pairs(pairs, dsu_tmp)
        if K > n*50 and len(pairs) < K:
            # safety: all pairs exhausted? shouldn't happen
            break
    # Now we know that the last needed edge is within first K pairs.
    # Let's find the exact last union edge using a fresh DSU and scan.
    dsu2 = DSU(n)
    last_edge = None
    for _, i, j in pairs:
        if dsu2.union(i, j):
            last_edge = (i, j)
            if dsu2.components == 1:
                break
    if last_edge is None:
        # Fallback: compute more pairs
        more = k_closest_pairs(points, K*2)
        dsu2 = DSU(n)
        for _, i, j in more:
            if dsu2.union(i, j):
                last_edge = (i, j)
                if dsu2.components == 1:
                    pairs = more
                    break
    if last_edge is None:
        raise RuntimeError("Failed to connect graph with streamed pairs")
    a, b = last_edge
    return points[a][0] * points[b][0], pairs, last_edge
