    #!/usr/bin/env python3
import math
from heapq import nsmallest

pts = []
with open('input.txt') as f:
    for line in f:
        line=line.strip()
        if not line: continue
        x,y,z = map(int,line.split(','))
        pts.append((x,y,z))
N=len(pts)
print('N=',N)

# generate all pair distances squared
import itertools

def gen_edges():
    for i in range(N):
        xi,yi,zi = pts[i]
        for j in range(i+1,N):
            xj,yj,zj = pts[j]
            dx=xi-xj; dy=yi-yj; dz=zi-zj
            d=dx*dx+dy*dy+dz*dz
            yield (d,i,j)

# get 1000 smallest edges
K=1000
edges = nsmallest(K, gen_edges())
# sort edges by distance
edges.sort()
print('Got',len(edges),'edges')

# Union-Find
parent=list(range(N))
size=[1]*N

def find(a):
    while parent[a]!=a:
        parent[a]=parent[parent[a]]
        a=parent[a]
    return a

def union(a,b):
    ra,rb=find(a),find(b)
    if ra==rb: return False
    if size[ra]<size[rb]: ra,rb=rb,ra
    parent[rb]=ra
    size[ra]+=size[rb]
    return True

# process edges (all 1000) performing union when possible
for d,i,j in edges:
    union(i,j)

# compute component sizes
roots={}
for i in range(N):
    r=find(i)
    roots[r]=roots.get(r,0)+1
sizes=sorted(roots.values(), reverse=True)
print('Sizes top 10',sizes[:10])
from functools import reduce
from operator import mul
while len(sizes)<3: sizes.append(1)
ans = sizes[0]*sizes[1]*sizes[2]
print('Answer',ans)

# save answer to file
with open('answer.txt','w') as f:
    f.write(str(ans)+"\n")

# create visualization: scatter 2D (x,y) and draw first 100 edges
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
xs=[p[0] for p in pts]
ys=[p[1] for p in pts]
plt.figure(figsize=(8,8))
plt.scatter(xs,ys,s=10)
for k,(d,i,j) in enumerate(edges[:100]):
    x1,y1,_=pts[i]; x2,y2,_=pts[j]
    plt.plot([x1,x2],[y1,y2],c='C0',alpha=0.3)
plt.title(f'N={N}, 1000 closest edges, show first 100')
plt.savefig('edges100.png')
print('Wrote edges100.png')
