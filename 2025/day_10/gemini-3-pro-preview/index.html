import sys

def solve():

        with open('input.txt', 'r') as f:
            lines = f.readlines()
    except FileNotFoundError:
        print("input.txt not found")
        return

    total_presses = 0

    for line in lines:
        line = line.strip()
        if not line:
            continue

        # Parse target
        match = re.match(r'\[([.#]+)\]', line)
        if not match:
            print(f"Failed to parse line: {line}")
            continue

        target_str = match.group(1)
        num_lights = len(target_str)
        target_mask = 0
        for i, char in enumerate(target_str):
            if char == '#':
                target_mask |= (1 << i)

        # Parse buttons
        # The remainder of the line after [...] contains (...) parts and {...}
        # We can just find all (...) parts

        button_matches = re.findall(r'\(([\d,]+)\)', line)
        buttons = []
        for b_str in button_matches:
            indices = [int(x) for x in b_str.split(',')]
            b_mask = 0
            for idx in indices:
                b_mask |= (1 << idx)
            buttons.append(b_mask)

        # Brute force solver
        num_buttons = len(buttons)
        min_presses = float('inf')
        found = False

        # Iterate all 2^num_buttons subsets
        # We can use recursion or bit manipulation

        # Optimization: limit search if we find a solution? 
        # But we need MINIMUM presses, so we might need to check others.
        # Actually, we can iterate by number of presses k = 0 to num_buttons
        # But iterating combinations(buttons, k) is better.

        from itertools import combinations

        for k in range(num_buttons + 1):
            # Try all combinations of length k
            # But wait, checking specific lengths first allows early exit!
            # If we find a solution with k presses, it's guaranteed minimal
            # because we check k=0, then k=1, etc.

            for combo in combinations(buttons, k):
                # sum (XOR) the masks
                current_mask = 0
                for b_mask in combo:
                    current_mask ^= b_mask

                if current_mask == target_mask:
                    min_presses = k
                    found = True
                    break
            if found:
                break

        if found:
            total_presses += min_presses
        else:
            print(f"No solution found for line: {line}")
            # Should we panic? Or just continue? 
            # Advent of Code usually guarantees solutions.

    print(f"Total minimum presses: {total_presses}")

    import re
import sys
from fractions import Fraction
import math

# Increase recursion depth just in case
sys.setrecursionlimit(2000)

def parse_input(filename):
    machines = []
    try:
        with open(filename, 'r') as f:
            lines = f.readlines()
    except FileNotFoundError:
        return []

    for line in lines:
        line = line.strip()
        if not line:
            continue

        # Parse target joltage { ... }
        match_target = re.search(r'\{([\d,]+)\}', line)
        if not match_target:
            continue
        targets = [int(x) for x in match_target.group(1).split(',')]

        # Parse buttons ( ... )
        # Find all (...) before the {...}
        # Actually just find all matches of parens
        button_matches = re.findall(r'\(([\d,]+)\)', line)
        buttons = []
        for b_str in button_matches:
            indices = [int(x) for x in b_str.split(',')]
            # Convert to vector
            # The length of vector is len(targets)
            vec = [0] * len(targets)
            for idx in indices:
                if idx < len(targets):
                    vec[idx] = 1
            buttons.append(vec)

        machines.append({'b': targets, 'A': buttons, 'orig_line': line})
    return machines

def solve_machine(machine):
    targets = machine['b'] # vector b of size R
    buttons = machine['A'] # list of cols, each size R. m buttons.

    R = len(targets)
    m = len(buttons)

    # Build augmented matrix for Ax = b
    # Matrix M has R rows and m+1 columns
    M = []
    for r in range(R):
        row = [Fraction(buttons[c][r]) for c in range(m)]
        row.append(Fraction(targets[r]))
        M.append(row)

    # Gaussian Elimination to RREF
    pivot_cols = []
    free_cols = []

    row_idx = 0
    pivots = {} # col -> row

    for c in range(m):
        # Find pivot
        pivot_row = -1
        for r in range(row_idx, R):
            if M[r][c] != 0:
                pivot_row = r
                break

        if pivot_row == -1:
            free_cols.append(c)
            continue

        pivot_cols.append(c)
        pivots[c] = row_idx

        # Swap rows
        M[row_idx], M[pivot_row] = M[pivot_row], M[row_idx]

        # Normalize row
        pivot_val = M[row_idx][c]
        for k in range(c, m + 1):
            M[row_idx][k] /= pivot_val

        # Eliminate other rows
        for r in range(R):
            if r != row_idx and M[r][c] != 0:
                factor = M[r][c]
                for k in range(c, m + 1):
                    M[r][k] -= factor * M[row_idx][k]

        row_idx += 1

    # Check consistency for remaining rows
    for r in range(row_idx, R):
        if M[r][m] != 0:
            # 0 = nonzero, impossible
            return float('inf')

    # Extract solution parameterization
    # x_p = Val - sum(coeff * x_f)
    # We want to express pivot variables in terms of free variables.

    # Base values for pivot vars (when free vars = 0)
    # Coeffs for free vars

    # x_p = constant_term + sum( free_term_k * x_fk )

    pivot_expr = {} # pivot_col -> (constant, {free_col: coeff})

    for p in pivot_cols:
        r = pivots[p]
        constant = M[r][m]
        coeffs = {}
        for f in free_cols:
            # In the row for p, we have: 1*x_p + ... + M[r][f]*x_f + ... = constant
            # x_p = constant - M[r][f]*x_f
            if M[r][f] != 0:
                coeffs[f] = -M[r][f]
        pivot_expr[p] = (constant, coeffs)

    # We need to find non-negative integers x_f such that x_p are non-negative integers
    # and sum(x) is minimized.

    # Objective function: sum(x_p) + sum(x_f)
    # = sum(constant_p + sum(coeff_pf * x_f)) + sum(x_f)
    # = sum(constant_p) + sum(x_f * (1 + sum(coeff_pf)))

    base_cost = sum(pivot_expr[p][0] for p in pivot_cols)

    cost_coeffs = {}
    for f in free_cols:
        c = Fraction(1)
        for p in pivot_cols:
            if f in pivot_expr[p][1]:
                c += pivot_expr[p][1][f]
        cost_coeffs[f] = c

    # Search over free variables
    # Constraints:
    # x_f >= 0, integer
    # x_p >= 0, integer => constant + sum(coeff * x_f) >= 0, integer

    # Upper bound for variables?
    # x_i <= max(targets) (since buttons add only positive)
    # We can calculate strict upper bound for each free variable?
    # Or just use safe bound.
    safe_bound = max(targets) + 1

    best_cost = float('inf')

    # Use simple recursion
    # Sort free cols?

    num_free = len(free_cols)
    current_free_vals = [0] * num_free

    def search(idx, current_cost_val):
        nonlocal best_cost

        # Pruning by cost?
        # If all future cost_coeffs are non-negative, and current_cost > best_cost, prune.
        # If some are negative, we can't easily prune by lower bound without looking ahead.
        # But let's check:
        # Lower bound estimation:
        # For remaining free vars, if cost_coeff > 0, min contribution is 0.
        # If cost_coeff < 0, min contribution is cost_coeff * max_bound.

        potential_min = current_cost_val
        for k in range(idx, num_free):
            f = free_cols[k]
            cc = cost_coeffs[f]
            if cc < 0:
                potential_min += cc * safe_bound 
                # This is a very loose bound, might not prune effectively but safe.

        if potential_min >= best_cost:
            return

        if idx == num_free:
            # Check integer constraints and non-negativity for pivots
            valid = True

            # We already maintain non-negativity of free vars by loop
            # Check pivots
            total_sum = 0

            # Map free col to val
            f_vals = {free_cols[k]: current_free_vals[k] for k in range(num_free)}

            # Verify pivots
            for p in pivot_cols:
                val = pivot_expr[p][0]
                for f, coeff in pivot_expr[p][1].items():
                    val += coeff * f_vals[f]

                if val < 0 or val.denominator != 1:
                    valid = False
                    break

            if valid:
                final_cost = current_cost_val 
                # Floating point/fraction comparison
                if final_cost < best_cost:
                    best_cost = final_cost
            return

        # Iterate current free var
        f = free_cols[idx]

        # Determine strict bounds for this variable?
        # x_p = C + coeff * x_f + ... >= 0
        # If coeff < 0, x_f <= (C + ...) / (-coeff)
        # If coeff > 0, x_f >= (-C - ...) / coeff

        # Since we don't know other vars yet, we can't get exact bounds.
        # But we can get bounds assuming others are 0 (if coeffs match signs)?
        # Let's just loop 0..safe_bound.
        # Optimize order: if cost_coeff < 0, loop high down to 0. Else 0 up.

        cc = cost_coeffs[f]

        loop_range = range(safe_bound)
        if cc < 0:
            loop_range = range(safe_bound - 1, -1, -1)

        for val in loop_range:
            current_free_vals[idx] = val
            new_cost = current_cost_val + cc * val

            # Intermediate Check:
            # Check if any pivot becomes strictly negative *impossible to recover*?
            # x_p = C + coeff_f * val + sum(coeff_k * x_k)
            # If coeff_k for all k > idx are <= 0, then we can't increase x_p.
            # So if current x_p < 0, and we can only decrease it further, prune.

            possible = True
            for p in pivot_cols:
                curr_p_val = pivot_expr[p][0]
                # Add contribution from current and past free vars
                for k in range(idx + 1):
                    f_prev = free_cols[k]
                    if f_prev in pivot_expr[p][1]:
                        curr_p_val += pivot_expr[p][1][f_prev] * current_free_vals[k]

                # Check potential future contribution
                # Max possible addition?
                max_add = 0
                for k in range(idx + 1, num_free):
                    f_next = free_cols[k]
                    if f_next in pivot_expr[p][1]:
                        c_next = pivot_expr[p][1][f_next]
                        if c_next > 0:
                            max_add += c_next * safe_bound

                if curr_p_val + max_add < 0:
                    possible = False
                    break

            if possible:
                search(idx + 1, new_cost)

            # If we just determined "impossible", do we break?
            # It depends on monotonicity.
            # If `cc` suggests direction, and we hit limit.

    search(0, base_cost)

    return best_cost

def solve():
    machines = parse_input('input.txt')
    total_presses = 0

    for i, m in enumerate(machines):
        res = solve_machine(m)
        # print(f"Machine {i}: {res}")
        if res != float('inf'):
            total_presses += res
        else:
            # Unsolvable machine?
            # Problem implies solution exists.
            pass

    print(f"Total minimum presses: {int(total_presses)}")

if __name__ == '__main__':
    solve()

    import matplotlib.pyplot as plt
import networkx as nx
import re

# Pick a complex machine to visualize
# Line 11 from input seems good
line_idx = 10
try:
    with open('input.txt', 'r') as f:
        lines = f.readlines()
        if len(lines) > line_idx:
            line = lines[line_idx].strip()
        else:
            line = lines[0].strip()
except:
    line = "[.##.] (3) (1,3) (2) (2,3) (0,2) (0,1) {3,5,4,7}"

# Parse
buttons = []
button_matches = re.findall(r'\(([\d,]+)\)', line)
for i, b_str in enumerate(button_matches):
    indices = [int(x) for x in b_str.split(',')]
    buttons.append((f"B{i}", indices))

# Create Bipartite Graph
G = nx.Graph()
button_nodes = [b[0] for b in buttons]
# Determine max counter index
max_idx = 0
for b in buttons:
    if b[1]:
        max_idx = max(max_idx, max(b[1]))
counter_nodes = [f"C{j}" for j in range(max_idx + 1)]

G.add_nodes_from(button_nodes, bipartite=0)
G.add_nodes_from(counter_nodes, bipartite=1)

edges = []
for b_name, indices in buttons:
    for idx in indices:
        edges.append((b_name, f"C{idx}"))
G.add_edges_from(edges)

# Draw
plt.figure(figsize=(10, 6))
pos = nx.bipartite_layout(G, button_nodes)
nx.draw_networkx_nodes(G, pos, nodelist=button_nodes, node_color='skyblue', node_size=500, label='Buttons')
nx.draw_networkx_nodes(G, pos, nodelist=counter_nodes, node_color='lightgreen', node_size=500, label='Counters')
nx.draw_networkx_edges(G, pos, edge_color='gray')
nx.draw_networkx_labels(G, pos)

plt.title("Wiring Schematic for Machine #11")
plt.axis('off')
plt.tight_layout()
plt.savefig('machine_graph.png')
