    import sys
from typing import List, Tuple, Dict
import bisect


def parse_input(path: str) -> List[Tuple[int, int]]:
    pts = []
    with open(path, 'r', encoding='utf-8') as f:
        for line in f:
            line = line.strip()
            if not line:
                continue
            x, y = line.split(',')
            pts.append((int(x), int(y)))
    return pts


def max_rectangle_area_inclusive(points: List[Tuple[int, int]]):
    n = len(points)
    best_area = 0
    best_pair = None
    pts = points
    # area in tiles: (|dx|+1)*(|dy|+1)
    for i in range(n):
        xi, yi = pts[i]
        for j in range(i + 1, n):
            xj, yj = pts[j]
            area = (abs(xi - xj) + 1) * (abs(yi - yj) + 1)
            if area > best_area:
                best_area = area
                best_pair = (i, j)
    return best_area, best_pair


def build_edges(ordered_points: List[Tuple[int, int]]):
    n = len(ordered_points)
    edges = []  # list of ((x1,y1),(x2,y2))
    for i in range(n):
        x1, y1 = ordered_points[i]
        x2, y2 = ordered_points[(i + 1) % n]
        # they guarantee axis-aligned adjacency
        edges.append(((x1, y1), (x2, y2)))
    return edges


def index_edges(edges: List[Tuple[Tuple[int, int], Tuple[int, int]]]):
    vert: Dict[int, List[Tuple[int, int]]] = {}
    horiz: Dict[int, List[Tuple[int, int]]] = {}
    for (x1, y1), (x2, y2) in edges:
        if x1 == x2:
            a, b = (y1, y2) if y1 <= y2 else (y2, y1)
            vert.setdefault(x1, []).append((a, b))
        elif y1 == y2:
            a, b = (x1, x2) if x1 <= x2 else (x2, x1)
            horiz.setdefault(y1, []).append((a, b))
        else:
            # Shouldn't happen
            raise ValueError("Non-orthogonal edge encountered")
    xs_sorted = sorted(vert.keys())
    ys_sorted = sorted(horiz.keys())
    return vert, horiz, xs_sorted, ys_sorted


def interval_overlap_open(a1: int, a2: int, b1: int, b2: int) -> bool:
    # Return True if open intervals (a1,a2) and (b1,b2) overlap with positive length
    lo = max(a1, b1)
    hi = min(a2, b2)
    return hi > lo


def rect_interior_intersects_boundary(xlo: int, xhi: int, ylo: int, yhi: int,
                                       vert, horiz, xs_sorted, ys_sorted) -> bool:
    # Check vertical edges at x in (xlo,xhi)
    li = bisect.bisect_right(xs_sorted, xlo)
    ri = bisect.bisect_left(xs_sorted, xhi)
    for idx in range(li, ri):
        x = xs_sorted[idx]
        for (ya, yb) in vert[x]:
            if interval_overlap_open(ya, yb, ylo, yhi):
                return True
    # Check horizontal edges at y in (ylo,yhi)
    li = bisect.bisect_right(ys_sorted, ylo)
    ri = bisect.bisect_left(ys_sorted, yhi)
    for idx in range(li, ri):
        y = ys_sorted[idx]
        for (xa, xb) in horiz[y]:
            if interval_overlap_open(xa, xb, xlo, xhi):
                return True
    return False


def point_in_polygon_inclusive(ptx: float, pty: float, poly: List[Tuple[int, int]]) -> bool:
    # Ray casting: count crossings to the right; treat boundary as inside.
    # Handle boundary check separately for horizontal and vertical segments.
    n = len(poly)
    inside = False
    for i in range(n):
        x1, y1 = poly[i]
        x2, y2 = poly[(i + 1) % n]
        # Boundary check: point on segment
        if x1 == x2:
            # vertical segment
            if abs(ptx - x1) < 1e-9 and min(y1, y2) - 1e-9 <= pty <= max(y1, y2) + 1e-9:
                return True
        elif y1 == y2:
            # horizontal segment
            if abs(pty - y1) < 1e-9 and min(x1, x2) - 1e-9 <= ptx <= max(x1, x2) + 1e-9:
                return True
        # Ray crossing test (standard, ignoring horizontal edges by convention)
        # We count edges where y1 > pty >= y2 or y2 > pty >= y1, then check if x at intersection > ptx
        y_low = y1
        y_high = y2
        x_at_y = None
        if ((y1 > pty) != (y2 > pty)):
            # Compute x coordinate of intersection of the edge with horizontal line at pty
            # Since edges are axis-aligned, this is easy
            if x1 == x2:
                x_at_y = x1
            else:
                # horizontal edge won't satisfy the condition above because y1==y2
                x_at_y = None
            if x_at_y is not None and x_at_y > ptx:
                inside = not inside
    return inside


def max_rectangle_area_part2(points: List[Tuple[int, int]], ordered_points: List[Tuple[int, int]]):
    # Build polygon boundary edges and indices
    edges = build_edges(ordered_points)
    vert, horiz, xs_sorted, ys_sorted = index_edges(edges)

    n = len(points)
    best_area = 0
    best_pair = None
    best_rect = None

    # Iterate over all pairs of red points as opposite corners
    for i in range(n):
        xi, yi = points[i]
        for j in range(i + 1, n):
            xj, yj = points[j]
            xlo = min(xi, xj)
            xhi = max(xi, xj)
            ylo = min(yi, yj)
            yhi = max(yi, yj)
            # Quick reject: degenerate to a line is still valid (width or height 0)
            # Check if boundary intersects rectangle interior
            if rect_interior_intersects_boundary(xlo, xhi, ylo, yhi, vert, horiz, xs_sorted, ys_sorted):
                continue
            # Test a point strictly inside rectangle interior (if interior area > 0)
            if xlo != xhi and ylo != yhi:
                cx = (xlo + xhi) / 2.0
                cy = (ylo + yhi) / 2.0
                if not point_in_polygon_inclusive(cx, cy, ordered_points):
                    continue
            else:
                # If it's a thin rectangle (line), test a point in its interior: use midpoint offset inside
                if xlo == xhi and ylo != yhi:
                    cx = xlo + 0.1
                    cy = (ylo + yhi) / 2.0
                elif ylo == yhi and xlo != xhi:
                    cx = (xlo + xhi) / 2.0
                    cy = ylo + 0.1
                else:
                    # single point (zero area rectangle), always allowed if that point is inside or on boundary
                    cx = xlo + 0.0
                    cy = ylo + 0.0
                if not point_in_polygon_inclusive(cx, cy, ordered_points):
                    continue
            area = (xhi - xlo + 1) * (yhi - ylo + 1)
            if area > best_area:
                best_area = area
                best_pair = (i, j)
                best_rect = (xlo, ylo, xhi, yhi)
    return best_area, best_pair, best_rect

def index_edges(edges: List[Tuple[Tuple[int, int], Tuple[int, int]]]):
    vert: Dict[int, List[Tuple[int, int]]] = {}
    horiz: Dict[int, List[Tuple[int, int]]] = {}
    for (x1, y1), (x2, y2) in edges:
        if x1 == x2:
            a, b = (y1, y2) if y1 <= y2 else (y2, y1)
            vert.setdefault(x1, []).append((a, b))
        elif y1 == y2:
            a, b = (x1, x2) if x1 <= x2 else (x2, x1)
            horiz.setdefault(y1, []).append((a, b))
        else:
            # Shouldn't happen
            raise ValueError("Non-orthogonal edge encountered")
    xs_sorted = sorted(vert.keys())
    ys_sorted = sorted(horiz.keys())
    return vert, horiz, xs_sorted, ys_sorted


def interval_overlap_open(a1: int, a2: int, b1: int, b2: int) -> bool:
    # Return True if open intervals (a1,a2) and (b1,b2) overlap with positive length
    lo = max(a1, b1)
    hi = min(a2, b2)
    return hi > lo


def rect_interior_intersects_boundary(xlo: int, xhi: int, ylo: int, yhi: int,
                                       vert, horiz, xs_sorted, ys_sorted) -> bool:
    # Check vertical edges at x in (xlo,xhi)
    li = bisect.bisect_right(xs_sorted, xlo)
    ri = bisect.bisect_left(xs_sorted, xhi)
    for idx in range(li, ri):
        x = xs_sorted[idx]
        for (ya, yb) in vert[x]:
            if interval_overlap_open(ya, yb, ylo, yhi):
                return True
    # Check horizontal edges at y in (ylo,yhi)
    li = bisect.bisect_right(ys_sorted, ylo)
    ri = bisect.bisect_left(ys_sorted, yhi)
    for idx in range(li, ri):
        y = ys_sorted[idx]
        for (xa, xb) in horiz[y]:
            if interval_overlap_open(xa, xb, xlo, xhi):
                return True
    return False


def point_in_polygon_inclusive(ptx: float, pty: float, poly: List[Tuple[int, int]]) -> bool:
    # Ray casting: count crossings to the right; treat boundary as inside.
    # Handle boundary check separately for horizontal and vertical segments.
    n = len(poly)
    inside = False
    for i in range(n):
        x1, y1 = poly[i]
        x2, y2 = poly[(i + 1) % n]
        # Boundary check: point on segment
        if x1 == x2:
            # vertical segment
            if abs(ptx - x1) < 1e-9 and min(y1, y2) - 1e-9 <= pty <= max(y1, y2) + 1e-9:
                return True
        elif y1 == y2:
            # horizontal segment
            if abs(pty - y1) < 1e-9 and min(x1, x2) - 1e-9 <= ptx <= max(x1, x2) + 1e-9:
                return True
        # Ray crossing test (standard, ignoring horizontal edges by convention)
        # We count edges where y1 > pty >= y2 or y2 > pty >= y1, then check if x at intersection > ptx
        y_low = y1
        y_high = y2
        x_at_y = None
        if ((y1 > pty) != (y2 > pty)):
            # Compute x coordinate of intersection of the edge with horizontal line at pty
            # Since edges are axis-aligned, this is easy
            if x1 == x2:
                x_at_y = x1
            else:
                # horizontal edge won't satisfy the condition above because y1==y2
                x_at_y = None
            if x_at_y is not None and x_at_y > ptx:
                inside = not inside
    return inside


def max_rectangle_area_part2(points: List[Tuple[int, int]], ordered_points: List[Tuple[int, int]]):
    # Build polygon boundary edges and indices
    edges = build_edges(ordered_points)
    vert, horiz, xs_sorted, ys_sorted = index_edges(edges)

    n = len(points)
    best_area = 0
    best_pair = None
    best_rect = None

    # Iterate over all pairs of red points as opposite corners
    for i in range(n):
        xi, yi = points[i]
        for j in range(i + 1, n):
            xj, yj = points[j]
            xlo = min(xi, xj)
            xhi = max(xi, xj)
            ylo = min(yi, yj)
            yhi = max(yi, yj)
            # Quick reject: degenerate to a line is still valid (width or height 0)
            # Check if boundary intersects rectangle interior
            if rect_interior_intersects_boundary(xlo, xhi, ylo, yhi, vert, horiz, xs_sorted, ys_sorted):
                continue
            # Test a point strictly inside rectangle interior (if interior area > 0)
            if xlo != xhi and ylo != yhi:
                cx = (xlo + xhi) / 2.0
                cy = (ylo + yhi) / 2.0
                if not point_in_polygon_inclusive(cx, cy, ordered_points):
                    continue
            else:
                # If it's a thin rectangle (line), test a point in its interior: use midpoint offset inside
                if xlo == xhi and ylo != yhi:
                    cx = xlo + 0.1
                    cy = (ylo + yhi) / 2.0
                elif ylo == yhi and xlo != xhi:
                    cx = (xlo + xhi) / 2.0
                    cy = ylo + 0.1
                else:
                    # single point (zero area rectangle), always allowed if that point is inside or on boundary
                    cx = xlo + 0.0
                    cy = ylo + 0.0
                if not point_in_polygon_inclusive(cx, cy, ordered_points):
                    continue
            area = (xhi - xlo + 1) * (yhi - ylo + 1)
            if area > best_area:
                best_area = area
                best_pair = (i, j)
                best_rect = (xlo, ylo, xhi, yhi)
    return best_area, best_pair, best_rect
