from bisect import bisect_right

# –í—Å—Ç—Ä–æ–∏–º —Å—é–¥–∞ –ø–∞—Ä—Å–µ—Ä –∏ –º–µ—Ä–∂ –∏–∑ —Ä–µ—à–µ–Ω–∏—è

def parse_input(text: str):
    top, bottom = text.strip().split("\n\n", 1)
    intervals = []
    for line in top.splitlines():
        a, b = line.strip().split('-')
        intervals.append((int(a), int(b)))
    values = [int(s.strip()) for s in bottom.splitlines() if s.strip()]
    return intervals, values


def merge_intervals(intervals):
    if not intervals:
        return []
    norm = [(min(l, r), max(l, r)) for l, r in intervals]
    norm.sort()
    merged = []
    cur_l, cur_r = norm[0]
    for l, r in norm[1:]:
        if l <= cur_r + 1:
            if r > cur_r:
                cur_r = r
        else:
            merged.append((cur_l, cur_r))
            cur_l, cur_r = l, r
    merged.append((cur_l, cur_r))
    return merged


def count_in_ranges(values, merged):
    if not merged:
        return 0
    starts = [l for l, _ in merged]
    ends = [r for _, r in merged]
    c = 0
    for v in values:
        i = bisect_right(starts, v) - 1
        if i >= 0 and v <= ends[i]:
            c += 1
    return c

def save_svg(filename, width, height, content):
    with open(filename, 'w', encoding='utf-8') as f:
        f.write(f'<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}">\n')
        f.write('<style> .lbl{font:12px sans-serif; fill:#333} </style>\n')
        f.write(content)
        f.write('\n</svg>')

# –õ–∏–Ω–µ–π–Ω–∞—è —à–∫–∞–ª–∞: –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ x –≤ –ø–∏–∫—Å–µ–ª–∏

def scale_builder(xmin, xmax, width, pad=40):
    if xmax == xmin:
        xmax = xmin + 1
    span = xmax - xmin
    def sx(x):
        return pad + (x - xmin) * (width - 2*pad) / span
    return sx

# –†–∏—Å—É–µ–º –∏–Ω—Ç–µ—Ä–≤–∞–ª—ã –ø–æ–ª–æ—Å–∫–∞–º–∏

def draw_intervals(bands, y, color, height=8):
    parts = []
    for l, r in bands:
        parts.append(f'<rect x="{l}" y="{y}" width="{max(1, r-l)}" height="{height}" fill="{color}" opacity="0.8" />')
    return '\n'.join(parts)
