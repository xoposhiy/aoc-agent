    import matplotlib.pyplot as plt
import numpy as np

def visualize_ranges_and_merge(fresh_ranges):
    fig, ax = plt.subplots(figsize=(12, 6))

    # Sort ranges for visualization consistency
    fresh_ranges.sort()

    y_offset = 0
    colors = plt.cm.viridis(np.linspace(0, 1, len(fresh_ranges)))

    # Plot initial ranges
    for i, (start, end) in enumerate(fresh_ranges):
        ax.hlines(y_offset, start, end, color=colors[i], linewidth=8, label=f'Range {start}-{end}')
        ax.text(start - 1, y_offset, f'{start}', va='center', ha='right', fontsize=8)

    sections = content.split('\n\n')

    range_lines = sections[0].split('\n')

    fresh_ranges.sort()

            merged_ranges.append([start, end])

    for start, end in merged_ranges:
        total_fresh_ids += (end - start + 1)
