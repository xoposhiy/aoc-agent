    # Advent of Code 2025 Day 5 - Cafeteria
# Part 1 solution
# Approach:
# - Parse the input into two sections: inclusive ranges (a-b) and query IDs.
# - Merge overlapping ranges to obtain a sorted list of disjoint intervals.
# - For each query ID, use binary search to check if it falls into any merged interval.
#   This yields O(R log R + Q log R) time, which is efficient.
# - Print the count as "Part 1: <answer>".

from bisect import bisect_right

def parse_input(path: str):
    with open(path, 'r', encoding='utf-8') as f:
        content = f.read().strip('\n')
    if '\n\n' in content:
        a, b = content.split('\n\n', 1)
    else:
        # Fallback: try to find first empty line
        lines = content.splitlines()
        sep = 0
        for i, line in enumerate(lines):
            if line.strip() == '':
                sep = i
                break
        a = '\n'.join(lines[:sep])
        b = '\n'.join(lines[sep+1:])
    ranges = []
    for line in a.splitlines():
        line = line.strip()
        if not line:
            continue
        if '-' not in line:
            # In case format deviates slightly, skip
            continue
        s, e = line.split('-', 1)
        start = int(s.strip())
        end = int(e.strip())
        if start > end:
            start, end = end, start
        ranges.append((start, end))
    ids = []
    for line in b.splitlines():
        line = line.strip()
        if not line:
            continue
        ids.append(int(line))
    return ranges, ids


def merge_intervals(ranges):
    if not ranges:
        return []
    ranges.sort()
    merged = []
    cs, ce = ranges[0]
    for s, e in ranges[1:]:
        if s <= ce:  # overlap (inclusive ranges)
            if e > ce:
                ce = e
        else:
            merged.append((cs, ce))
            cs, ce = s, e
    merged.append((cs, ce))
    return merged


def count_inside(merged, values):
    # Prepare starts list for bisect
    starts = [s for s, _ in merged]
    count = 0
    for v in values:
        i = bisect_right(starts, v) - 1
        if i >= 0 and merged[i][0] <= v <= merged[i][1]:
            count += 1
    return count


def main():
    ranges, ids = parse_input('input.txt')
    merged = merge_intervals(ranges)
    ans1 = count_inside(merged, ids)
    print(f"Part 1: {ans1}")

if __name__ == '__main__':
    main()

    # Advent of Code 2025 Day 5: Cafeteria
# Idea:
# - Input has inclusive fresh ID ranges, blank line, then some query IDs (Part 1 only).
# - Merge overlapping ranges once. For Part 1, count how many query IDs fall into any merged range via binary search.
# - For Part 2, sum coverage of merged ranges: sum(r-l+1).
# Complexity: O(R log R + Q log R), with small constants.

from bisect import bisect_right
from typing import List, Tuple

Interval = Tuple[int, int]


def parse_input(path: str) -> tuple[list[Interval], list[int]]:
    with open(path, 'r', encoding='utf-8') as f:
        content = f.read().rstrip('\n')
    if '\n\n' in content:
        a, b = content.split('\n\n', 1)
    else:
        lines = content.splitlines()
        sep = 0
        for i, line in enumerate(lines):
            if line.strip() == '':
                sep = i
                break
        a = '\n'.join(lines[:sep])
        b = '\n'.join(lines[sep+1:]) if sep+1 < len(lines) else ''
    ranges: list[Interval] = []
    for line in a.splitlines():
        line = line.strip()
        if not line or '-' not in line:
            continue
        s, e = line.split('-', 1)
        l, r = int(s), int(e)
        if l > r:
            l, r = r, l
        ranges.append((l, r))
    ids: list[int] = []
    for line in b.splitlines():
        line = line.strip()
        if not line:
            continue
        ids.append(int(line))
    return ranges, ids

        else:
            merged.append((cs, ce))
            cs, ce = s, e
    merged.append((cs, ce))
    return merged


def count_inside(merged: List[Interval], values: List[int]) -> int:
    starts = [s for s, _ in merged]
    cnt = 0
    for v in values:
        i = bisect_right(starts, v) - 1
        if i >= 0 and merged[i][0] <= v <= merged[i][1]:
            cnt += 1
    return cnt


def sum_coverage(merged: List[Interval]) -> int:
    return sum(r - l + 1 for l, r in merged)


def main() -> None:
    ranges, ids = parse_input('input.txt')
    merged = merge_intervals(ranges)
    part1 = count_inside(merged, ids)
    part2 = sum_coverage(merged)
    print(f"Part 1: {part1}")
    print(f"Part 2: {part2}")

if __name__ == '__main__':
    main()
