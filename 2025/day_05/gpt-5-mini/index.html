- Part 1: count of available ids that fall into any merged interval.
- Part 2: total count of integers covered by merged intervals.

from bisect import bisect_right


def read_sections(path='input.txt'):
    with open(path,'r',encoding='utf-8') as f:
        text = f.read()
    parts = text.split('\n\n')
    ranges = parts[0].strip().splitlines() if parts and parts[0].strip() else []
    ids = []
    if len(parts) > 1:
        ids = parts[1].strip().splitlines() if parts[1].strip() else []

def parse_ranges(lines):
    intervals = []
    for ln in lines:
        ln=ln.strip()
        if not ln: continue
        a,b = ln.split('-')
        intervals.append((int(a), int(b)))
    return intervals


def merge_intervals(intervals):
    if not intervals: return []
    intervals.sort()
    merged = []
    s,e = intervals[0]
    for ns,ne in intervals[1:]:
        if ns <= e + 1:
            e = max(e, ne)
        else:
            merged.append((s,e))
            s,e = ns,ne
    merged.append((s,e))
    return merged


def count_available_in_merged(merged, ids):
    starts = [s for s,e in merged]
    cnt = 0
    for ln in ids:
        if not ln: continue
        x = int(ln)
        i = bisect_right(starts, x) - 1
        if i >= 0 and merged[i][0] <= x <= merged[i][1]:
            cnt += 1
    return cnt

def count_fresh(merged, ids):
    starts = [s for s,e in merged]
    cnt = 0
    for id_s in ids:
        if not id_s: continue
        x = int(id_s)
        # find rightmost interval with start <= x
        i = bisect_right(starts, x) - 1
        if i >= 0 and merged[i][0] <= x <= merged[i][1]:
            cnt += 1
    return cnt
