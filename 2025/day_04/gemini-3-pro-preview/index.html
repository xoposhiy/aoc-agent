    def solve():
    with open('input.txt', 'r') as f:
        grid = [list(line.strip()) for line in f if line.strip()]

    rows = len(grid)
    cols = len(grid[0])

    # Store neighbor counts for all rolls
    neighbor_counts = {}
    rolls = set()

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '@':
                rolls.add((r, c))

    # Calculate initial neighbor counts
    queue = []

    for r, c in rolls:
        count = 0
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr == 0 and dc == 0:
                    continue
                nr, nc = r + dr, c + dc
                if (nr, nc) in rolls:
                    count += 1
        neighbor_counts[(r, c)] = count
        if count < 4:
            queue.append((r, c))

    removed = set()
    removed_count = 0

    # Process queue
    # Using an index for queue instead of popping from front for efficiency if simple list, 
    # but Python list pop(0) is O(k), collections.deque is O(1). 
    # Since order doesn't strictly matter for the final count, list as stack is fine?
    # Actually, order might matter for the sequence, but not for the final set of removed nodes 
    # because the property is monotonic.
    # Let's use a list and pop() (DFS-like) or deque (BFS-like). 
    # The problem asks for total count, so order doesn't matter.

    import collections
    q = collections.deque(queue)

    while q:
        r, c = q.popleft()

        if (r, c) in removed:
            continue

        removed.add((r, c))
        removed_count += 1

        # Update neighbors
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr == 0 and dc == 0:
                    continue
                nr, nc = r + dr, c + dc

                if (nr, nc) in rolls and (nr, nc) not in removed:
                    neighbor_counts[(nr, nc)] -= 1
                    if neighbor_counts[(nr, nc)] == 3:
                        q.append((nr, nc))

def solve():
    # Read input
    with open('input.txt', 'r') as f:
        grid = [list(line.strip()) for line in f if line.strip()]

    rows = len(grid)
    cols = len(grid[0])

    # Identify all paper rolls
    rolls = set()
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '@':
                rolls.add((r, c))

    # Precompute neighbors for graph
    # neighbor_map: (r, c) -> list of neighbor coords
    neighbor_map = collections.defaultdict(list)
    for r, c in rolls:
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr == 0 and dc == 0:
                    continue
                nr, nc = r + dr, c + dc
                if (nr, nc) in rolls:
                    neighbor_map[(r, c)].append((nr, nc))

    # Part 1: Count initially accessible rolls
    initial_accessible = 0
    # Also prepare for Part 2
    neighbor_counts = {}
    queue = collections.deque()

    for pos in rolls:
        count = len(neighbor_map[pos])
        neighbor_counts[pos] = count
        if count < 4:
            initial_accessible += 1
            queue.append(pos)

    print(f"Part 1: {initial_accessible}")

    # Part 2: Simulation
    removed = {} # Store removal order (index)
    removal_sequence = []

    while queue:
        pos = queue.popleft()

        if pos in removed:
            continue

        removed[pos] = len(removal_sequence)
        removal_sequence.append(pos)

        for neighbor in neighbor_map[pos]:
            if neighbor not in removed:
                neighbor_counts[neighbor] -= 1
                if neighbor_counts[neighbor] == 3:
                    queue.append(neighbor)

    print(f"Part 2: {len(removed)}")

def generate_heatmap(rows, cols, rolls, removed, max_order):
    # Generates an SVG heatmap
    cell_size = 5
    width = cols * cell_size
    height = rows * cell_size

    svg_content = [f'<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}" style="background-color:black">']

    # Draw non-removed rolls as gray
    for r, c in rolls:
        if (r, c) not in removed:
            x = c * cell_size
            y = r * cell_size
            svg_content.append(f'<rect x="{x}" y="{y}" width="{cell_size}" height="{cell_size}" fill="#333" />')

    # Draw removed rolls with color gradient
    # Gradient from Blue (start) to Red (end) via Green/Yellow?
    # Or HSV hue rotation.

    for (r, c), order in removed.items():
        if max_order > 1:
            ratio = order / (max_order - 1)
        else:
            ratio = 0

        # Hue: 240 (blue) to 0 (red)
        hue = 240 * (1 - ratio)
        # Convert to RGB
        # h in [0, 360], s=1, v=1

        # Simple hsv_to_rgb implementation
        h = hue / 360.0
        s = 1.0
        v = 1.0

        i = int(h * 6)
        f = h * 6 - i
        p = v * (1 - s)
        q = v * (1 - f * s)
        t = v * (1 - (1 - f) * s)

        if i % 6 == 0: r_val, g_val, b_val = v, t, p
        elif i % 6 == 1: r_val, g_val, b_val = q, v, p
        elif i % 6 == 2: r_val, g_val, b_val = p, v, t
        elif i % 6 == 3: r_val, g_val, b_val = p, q, v
        elif i % 6 == 4: r_val, g_val, b_val = t, p, v
        else: r_val, g_val, b_val = v, p, q

        color = f"rgb({int(r_val*255)},{int(g_val*255)},{int(b_val*255)})"

        x = c * cell_size
        y = r * cell_size
        svg_content.append(f'<rect x="{x}" y="{y}" width="{cell_size}" height="{cell_size}" fill="{color}" />')

    svg_content.append('</svg>')

    with open('heatmap.svg', 'w') as f:
        f.write("\n".join(svg_content))
