    import sys

def parse_input(filename):
    with open(filename, 'r') as f:
        content = f.read()

    parts = content.split('\n\n')

    shapes = {}
    regions_part = []

    for part in parts:
        lines = part.strip().split('\n')
        if not lines: continue

        # Check if it's a shape
        if ':' in lines[0] and not 'x' in lines[0].split(':')[0]:
            idx = int(lines[0].split(':')[0])
            grid = [list(row) for row in lines[1:]]
            shapes[idx] = grid
        else:
            # It's regions
            # But the input has regions line by line?
            # Let's check if the part contains multiple regions or just one
            # The input format: 
            # 38x36: ...
            # 48x43: ...
            # So the regions might be in one big block at the end?
            # In the provided input.txt, the shapes come first, then a long list of regions.
            # But in the read_file output, it looks like regions are just lines.
            # The split('\n\n') might split the shapes, and then the last part is the block of regions.

            # Let's handle lines that look like region definitions
            for line in lines:
                if 'x' in line and ':' in line:
                    regions_part.append(line)

    return shapes, regions_part

def get_shape_size(shape):
    return sum(row.count('#') for row in shape)

def solve():
    shapes, region_lines = parse_input('input.txt')

    shape_sizes = {idx: get_shape_size(grid) for idx, grid in shapes.items()}

    total_regions = 0
    area_fail = 0
    area_pass = 0

    pass_ratios = []

    for line in region_lines:
        total_regions += 1

def parse_input(filename):

def get_shape_size(shape):
    return sum(row.count('#') for row in shape)

def get_variants(shape):
    # Generate all 8 symmetries
    variants = []
    curr = shape
    for _ in range(2): # Flip
        for _ in range(4): # Rotate
            # Normalize
            # Find bounding box
            min_r, max_r = 100, -1
            min_c, max_c = 100, -1
            h = len(curr)
            w = len(curr[0])
            points = []
            for r in range(h):
                for c in range(w):
                    if curr[r][c] == '#':
                        points.append((r,c))
                        min_r = min(min_r, r)
                        max_r = max(max_r, r)
                        min_c = min(min_c, c)
                        max_c = max(max_c, c)

            # Shift to 0,0
            norm_points = set()
            norm_h = max_r - min_r + 1
            norm_w = max_c - min_c + 1
            grid = [['.' for _ in range(norm_w)] for _ in range(norm_h)]
            for r, c in points:
                nr, nc = r - min_r, c - min_c
                norm_points.add((nr, nc))
                grid[nr][nc] = '#'

            variants.append((norm_h, norm_w, norm_points))

            # Rotate 90 deg clockwise
            # (r, c) -> (c, h-1-r)
            new_h, new_w = w, h
            new_curr = [['.' for _ in range(new_w)] for _ in range(new_h)]
            for r in range(h):
                for c in range(w):
                    if curr[r][c] == '#':
                        new_curr[c][h-1-r] = '#'
            curr = new_curr

        # Flip
        curr = curr[::-1]

    # Deduplicate
    unique_variants = []
    seen = set()
    for h, w, pts in variants:
        # pts is a set of tuples, frozenset to hash
        fs = frozenset(pts)
        if fs not in seen:
            seen.add(fs)
            unique_variants.append((h, w, list(pts)))
    return unique_variants

def solve_one(w, h, shapes_to_place, shape_variants):
    # shapes_to_place is a list of shape_indices
    # Randomized greedy

    # Grid: false = empty, true = filled
    # use 1D array for speed? or just set of filled cells?
    # Set of filled cells is easier
    filled = set()

    # Try to place
    for idx in shapes_to_place:
        placed = False
        # Try all variants
        variants = shape_variants[idx]
        # Shuffle variants?
        # variants = list(variants)
        # random.shuffle(variants)

        # Try random positions? 
        # scanning all positions is slow if we do it every time.
        # But maybe we can find ANY valid spot.

        # To optimize: maintain a list of available cells?
        # Or just scan. 40x40 = 1600.

        # Heuristic: try to place near top-left?

        possible_moves = []

        # We need to find ONE valid placement.
        # If we check all positions for all variants, that's expensive.
        # Let's just scan linearly and take the first one.

        for v_h, v_w, v_pts in variants:
            # check limits
            if v_h > h or v_w > w: continue

            for r in range(h - v_h + 1):
                for c in range(w - v_w + 1):
                    # Check collision
                    ok = True
                    for pr, pc in v_pts:
                        if (r + pr, c + pc) in filled:
                            ok = False
                            break
                    if ok:
                        # Found a spot
                        # place it
                        for pr, pc in v_pts:
                            filled.add((r + pr, c + pc))
                        placed = True
                        break
                if placed: break
            if placed: break

        if not placed:
            return False

    return True

def main():
    shapes, region_lines = parse_input('input.txt')

    shape_sizes = {idx: get_shape_size(grid) for idx, grid in shapes.items()}

    shape_variants = {idx: get_variants(grid) for idx, grid in shapes.items()}

    # Find max density region
    max_ratio = -1
    target_region = None

    for line in region_lines:

            if ratio > max_ratio:
                max_ratio = ratio
                target_region = (rw, rh)
                target_counts = counts

    print(f"Max density region: {target_region} with ratio {max_ratio:.4f}")

    # Prepare list of shapes
    shapes_list = []
    for idx, cnt in enumerate(target_counts):
        shapes_list.extend([idx] * cnt)

    print(f"Number of shapes: {len(shapes_list)}")

    # Try to solve with randomized greedy
    # 100 tries
    success = False
    start_time = time.time()
    for attempt in range(100):

    ax.add_patch(patches.Rectangle((0,0), w, h, fill=False, edgecolor='black', linewidth=2))

    colors = ['#e6194b', '#3cb44b', '#ffe119', '#4363d8', '#f58231', '#911eb4']
    filenames = []

    def save_frame(idx):
        fname = f'frame_{idx:03d}.png'
        plt.savefig(fname, bbox_inches='tight', dpi=80)
        filenames.append(fname)

    save_frame(0)

    count = 0
    # Randomize order
    shapes_list = list(shapes_to_place)
    random.shuffle(shapes_list)

    # We need to backtrack or restart if fail, but for animation let's assume success or just show progress
    # To be safe, we just run the greedy pass once. If it fails, we restart.
    # But for animation, we want the successful run.

    # Let's first find a solution, then animate it.
    # Re-using solve logic

    solution = None
    while solution is None:
        random.shuffle(shapes_list)
        temp_filled = set()
        temp_placements = []
        possible = True

        for idx in shapes_list:
            placed = False
            vars_list = list(shape_variants[idx])
            random.shuffle(vars_list)

            for v_h, v_w, v_pts in vars_list:
                if v_h > h or v_w > w: continue
                for r in range(h - v_h + 1):
                    for c in range(w - v_w + 1):
                        ok = True
                        for pr, pc in v_pts:
                            if (r + pr, c + pc) in temp_filled:
                                ok = False
                                break
                        if ok:
                            pts = []
                            for pr, pc in v_pts:
                                temp_filled.add((r + pr, c + pc))
                                pts.append((r+pr, c+pc))
                            temp_placements.append((idx, pts))
                            placed = True
                            break
                    if placed: break
                if placed: break

            if not placed:
                possible = False
                break

        if possible:
            solution = temp_placements

    print("Solution found, generating frames...")

    # Now animate the solution
    for i, (idx, pts) in enumerate(solution):
        color = colors[idx % len(colors)]
        for r, c in pts:
            x = c
            y = h - 1 - r
            rect = patches.Rectangle((x, y), 1, 1, linewidth=0, edgecolor=None, facecolor=color)
            ax.add_patch(rect)

        if i % 5 == 0 or i == len(solution) - 1:
            save_frame(len(filenames))

    plt.close()

    # Create gif
    with imageio.get_writer('packing.gif', mode='I', duration=0.1) as writer:
        for filename in filenames:
            image = imageio.imread(filename)
            writer.append_data(image)

    # Cleanup
    for filename in filenames:
        os.remove(filename)

    print("GIF saved to packing.gif")

def main():
    shapes, region_lines = parse_input('input.txt')
    shape_sizes = {idx: get_shape_size(grid) for idx, grid in shapes.items()}
    shape_variants = {idx: get_variants(grid) for idx, grid in shapes.items()}

    # Max density region
    max_ratio = -1
    target_region = None
    target_counts = None

    for line in region_lines:
        dims, counts_str = line.split(':')
        rw, rh = map(int, dims.split('x'))
        region_area = rw * rh
        counts = list(map(int, counts_str.strip().split()))
        presents_area = sum(counts[i] * shape_sizes[i] for i in range(len(counts)))
        if presents_area <= region_area:
            ratio = presents_area / region_area
            if ratio > max_ratio:
                max_ratio = ratio
                target_region = (rw, rh)
                target_counts = counts

    shapes_list = []
    for idx, cnt in enumerate(target_counts):
        shapes_list.extend([idx] * cnt)

    print(f"Animating {target_region} with {len(shapes_list)} items...")
    solve_and_animate(target_region[0], target_region[1], shapes_list, shape_variants)

if __name__ == '__main__':
    main()
