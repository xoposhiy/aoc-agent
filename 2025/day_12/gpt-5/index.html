def count_ok(areas, regions):
    ok=0
    stats=[]
    for W,H,counts in regions:
        need = sum(c*a for c,a in zip(counts, areas))
        slack = W*H - need
        stats.append((W,H,need,slack))
        if slack >= 0:
            ok += 1
    return ok, stats
