    import sys
from bisect import bisect_left
from collections import deque


def parse_grid(lines):
    grid = [list(line.rstrip('\n')) for line in lines if line.strip('\n')]
    h = len(grid)
    w = len(grid[0]) if h else 0
    s_pos = None
    for r in range(h):
        for c in range(w):
            if grid[r][c] == 'S':
                s_pos = (r, c)
                break
        if s_pos:
            break
    if s_pos is None:
        raise ValueError("No S found")
    # Precompute splitters per column (sorted by row)
    cols = [[] for _ in range(w)]
    for r in range(h):
        for c in range(w):
            if grid[r][c] == '^':
                cols[c].append(r)
    return grid, s_pos, cols, h, w


def count_splits(grid, s_pos, cols, h, w):
    rS, cS = s_pos
    q = deque()
    processed = set()  # (col, idx of splitter in that column)
    # Beam starts just below S
    if rS + 1 < h:
        q.append((rS + 1, cS))
    splits = 0
    while q:
        r0, c = q.popleft()
        split_rows = cols[c]
        if not split_rows:
            continue
        idx = bisect_left(split_rows, r0)
        if idx >= len(split_rows):
            continue
        key = (c, idx)
        if key in processed:
            continue
        processed.add(key)
        r_split = split_rows[idx]
        splits += 1
        # spawn beams at same row from left and right columns
        if c - 1 >= 0:
            q.append((r_split, c - 1))
        if c + 1 < w:
            q.append((r_split, c + 1))
    return splits


def main():
    with open('input.txt', 'r', encoding='utf-8') as f:
        data = f.read().splitlines()
    grid, s_pos, cols, h, w = parse_grid(data)
    ans1 = count_splits(grid, s_pos, cols, h, w)
    print(ans1)

if __name__ == '__main__':
    main()

def parse_grid(lines):

    h = len(grid)
    w = len(grid[0]) if h else 0
    s_pos = None
    for r in range(h):
        for c in range(w):
            if grid[r][c] == 'S':
                s_pos = (r, c)
                break
        if s_pos:
            break
    if s_pos is None:
        raise ValueError("No S found")
    cols = [[] for _ in range(w)]
    for r in range(h):
        for c in range(w):
            if grid[r][c] == '^':
                cols[c].append(r)
    return grid, s_pos, cols, h, w


def count_timelines(grid, s_pos, cols, h, w):
    rS, cS = s_pos
    # Map each splitter node (c,i) -> number of terminal timelines from that node
    memo = {}
    sys.setrecursionlimit(1000000)

    def compute_f(start_c, start_i):
        # iterative DFS postorder to avoid recursion limits
        stack = [(start_c, start_i, 0)]
        while stack:
            c, i, stage = stack.pop()
            if (c, i) in memo:
                continue
            rows_c = cols[c]
            if stage == 0:
                stack.append((c, i, 1))
                r_split = rows_c[i]
                # left child
                nc = c - 1
                if nc >= 0:
                    rows = cols[nc]
                    j = bisect_left(rows, r_split)
                    if j < len(rows) and (nc, j) not in memo:
                        stack.append((nc, j, 0))
                # right child
                nc = c + 1
                if nc < w:
                    rows = cols[nc]
                    j = bisect_left(rows, r_split)
                    if j < len(rows) and (nc, j) not in memo:
                        stack.append((nc, j, 0))
            else:
                r_split = rows_c[i]
                total = 0
                # left
                nc = c - 1
                if nc < 0:
                    total += 1
                else:
                    rows = cols[nc]
                    j = bisect_left(rows, r_split)
                    if j == len(rows):
                        total += 1
                    else:
                        total += memo[(nc, j)]
                # right
                nc = c + 1
                if nc >= w:
                    total += 1
                else:
                    rows = cols[nc]
                    j = bisect_left(rows, r_split)
                    if j == len(rows):
                        total += 1
                    else:
                        total += memo[(nc, j)]
                memo[(c, i)] = total
        return memo[(start_c, start_i)]

    # initial beam starts below S
    rows = cols[cS]
    j0 = bisect_left(rows, rS + 1)
    if j0 == len(rows):
        return 1
    return compute_f(cS, j0)


def main():
    with open('input.txt', 'r', encoding='utf-8') as f:
        data = f.read().splitlines()
    grid, s_pos, cols, h, w = parse_grid(data)
    ans2 = count_timelines(grid, s_pos, cols, h, w)
    print(ans2)

if __name__ == '__main__':
    main()
