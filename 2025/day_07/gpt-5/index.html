    #!/usr/bin/env python3
import sys
from pathlib import Path

def parse_grid(text: str):
    lines = [list(line.rstrip('\n')) for line in text.splitlines() if line.strip('\n')]
    R = len(lines)
    C = len(lines[0]) if R else 0
    s_pos = None
    for r,row in enumerate(lines):
        for c,ch in enumerate(row):
            if ch == 'S':
                s_pos = (r,c)
    if s_pos is None:
        raise ValueError('No S found')
    return lines, s_pos, R, C


def simulate(lines, s_pos):
    R = len(lines)
    C = len(lines[0]) if R else 0
    sr, sc = s_pos
    active = set()
    # Start immediately below S
    if sr + 1 < R:
        active.add(sc)
    splits = 0
    passed = [[False]*C for _ in range(R)]
    split_hits = []  # list of (r,c) where split occurred
    for r in range(sr+1, R):
        next_active = set()
        # mark passes for visualization
        for c in active:
            if 0 <= c < C:
                passed[r][c] = True
        for c in active:
            if not (0 <= c < C):
                continue
            ch = lines[r][c]
            if ch == '.':
                next_active.add(c)
            elif ch == '^':
                splits += 1
                split_hits.append((r,c))
                if c-1 >= 0:
                    next_active.add(c-1)
                if c+1 < C:
                    next_active.add(c+1)
            elif ch == 'S':
                # unexpected below, but treat as empty
                next_active.add(c)
            else:
                # Any other symbols? Treat as empty to be safe
                next_active.add(c)
        active = next_active
        if not active:
            # all beams exited horizontally or were stopped
            # but they only move downward so they can only exit by empty next row not existing
            # Keep going until bottom anyway; active empty is terminal state
            # break  # But we must not break because visualization for rows below would be empty anyway
            pass
    return splits, passed, split_hits


def solve(text: str):
    lines, s_pos, R, C = parse_grid(text)
    splits, passed, split_hits = simulate(lines, s_pos)
    return splits, passed, split_hits, (R,C)


def main():
    input_path = Path('input.txt')
    data = input_path.read_text().rstrip('\n')
    part1, passed, split_hits, (R,C) = solve(data)
    print(part1)

if __name__ == '__main__':
    main()

    #!/usr/bin/env python3
import sys
from pathlib import Path

def parse_grid(text: str):
    lines = [list(line.rstrip('\n')) for line in text.splitlines() if line.strip('\n')]
    R = len(lines)
    C = len(lines[0]) if R else 0
    s_pos = None
    for r,row in enumerate(lines):
        for c,ch in enumerate(row):
            if ch == 'S':
                s_pos = (r,c)
    if s_pos is None:
        raise ValueError('No S found')
    return lines, s_pos, R, C


def count_timelines(lines, s_pos):
    R = len(lines)
    C = len(lines[0]) if R else 0
    sr, sc = s_pos
    # DP: for each row, a dict col->count ways to be at that col right above this row
    # Start with a single particle entering below S at column sc
    ways = {sc: 1}
    for r in range(sr+1, R):
        next_ways = {}
        for c, cnt in ways.items():
            if not (0 <= c < C):
                continue
            ch = lines[r][c]
            if ch == '.':
                next_ways[c] = next_ways.get(c, 0) + cnt
            elif ch == '^':
                if c-1 >= 0:
                    next_ways[c-1] = next_ways.get(c-1, 0) + cnt
                if c+1 < C:
                    next_ways[c+1] = next_ways.get(c+1, 0) + cnt
            elif ch == 'S':
                next_ways[c] = next_ways.get(c, 0) + cnt
            else:
                next_ways[c] = next_ways.get(c, 0) + cnt
        ways = next_ways
        if not ways:
            break
    # When we leave the last processed row, particle exits manifold; each distinct path is a timeline.
    # The total number of timelines is the sum over counts in last ways after processing last row inside grid
    # Actually, after last row processed (r==R-1), the particle exits. The number of timelines equals sum of ways.
    total = sum(ways.values())
    return total


def solve(text: str):
    lines, s_pos, R, C = parse_grid(text)
    return count_timelines(lines, s_pos)


def main():
    input_path = Path('input.txt')
    data = input_path.read_text().rstrip('\n')
    ans = solve(data)
    print(ans)

if __name__ == '__main__':
    main()

    #!/usr/bin/env python3
from pathlib import Path
import matplotlib.pyplot as plt
import numpy as np


def parse_grid(text: str):
    lines = [list(line.rstrip('\n')) for line in text.splitlines() if line.strip('\n')]
    R = len(lines)
    C = len(lines[0]) if R else 0
    s_pos = None
    for r,row in enumerate(lines):
        for c,ch in enumerate(row):
            if ch == 'S':
                s_pos = (r,c)
    if s_pos is None:
        raise ValueError('No S found')
    return lines, s_pos, R, C


def simulate_passed(lines, s_pos):
    R = len(lines)
    C = len(lines[0]) if R else 0
    sr, sc = s_pos
    active = set()
    if sr + 1 < R:
        active.add(sc)
    passed = [[False]*C for _ in range(R)]
    split_hits = []
    for r in range(sr+1, R):
        next_active = set()
        for c in active:
            if 0 <= c < C:
                passed[r][c] = True
        for c in active:
            if not (0 <= c < C):
                continue
            ch = lines[r][c]
            if ch == '.':
                next_active.add(c)
            elif ch == '^':
                split_hits.append((r,c))
                if c-1 >= 0:
                    next_active.add(c-1)
                if c+1 < C:
                    next_active.add(c+1)
            else:
                next_active.add(c)
        active = next_active
        if not active:
            break
    return passed, split_hits


def make_visual(input_file='input.txt', out_file='visualization.png'):
    text = Path(input_file).read_text().rstrip('\n')
    lines, s_pos, R, C = parse_grid(text)
    passed, split_hits = simulate_passed(lines, s_pos)

    # Build an image: 0 background, 1 passed, 2 splitter, 3 S
    img = np.zeros((R, C), dtype=np.uint8)
    for r in range(R):
        for c in range(C):
            if lines[r][c] == '^':
                img[r, c] = 2
            elif lines[r][c] == 'S':
                img[r, c] = 3
            elif passed[r][c]:
                img[r, c] = 1
            else:
                img[r, c] = 0
    # Custom colormap via ListedColormap-like manual mapping
    # We'll map: 0->white, 1->skyblue, 2->crimson, 3->gold
    colors = np.array([
        [1.0, 1.0, 1.0],      # background
        [0.53, 0.81, 0.98],   # beams
        [0.86, 0.08, 0.24],   # splitters
        [1.0, 0.84, 0.0],     # S
    ])
    rgb = colors[img]

    # Draw
    scale = max(6, min(18, 1200 // max(1, max(R, C))))
    figsize = (C/10, R/10)
    plt.figure(figsize=figsize, dpi=100)
    plt.imshow(rgb, interpolation='nearest')
    plt.title(' :      ,   ,    S', fontsize=10)
    plt.axis('off')

    # annotate a few split points to avoid clutter
    if split_hits:
        for (r,c) in split_hits[:min(30, len(split_hits))]:
            plt.text(c, r, '', color='black', ha='center', va='center', fontsize=6, alpha=0.7)

    plt.tight_layout(pad=0.2)
    plt.savefig(out_file, bbox_inches='tight', pad_inches=0.02)
    plt.close()

if __name__ == '__main__':
    make_visual()
