    #!/usr/bin/env python3
import sys
from pathlib import Path

def read_input() -> str:
    p = Path('input.txt')
    if p.exists():
        return p.read_text()
    # fallback to stdin (for local runs)
    return sys.stdin.read()


def parse_grid(text: str):
    grid = [list(line.rstrip('\n')) for line in text.splitlines() if line.strip('\n') != '']
    H = len(grid)
    W = len(grid[0]) if H else 0
    rS = cS = None
    for r in range(H):
        for c in range(W):
            if grid[r][c] == 'S':
                rS, cS = r, c
                break
        if rS is not None:
            break
    return grid, H, W, rS, cS


def part1(grid, H, W, rS, cS):
    # simulate row by row starting below S
    splits = 0
    if rS is None:
        return 0
    beams = set()
    if rS + 1 < H:
        beams.add(cS)
    for r in range(rS + 1, H):
        next_beams = set()
        for c in beams:
            cell = grid[r][c]
            if cell == '.':
                next_beams.add(c)
            elif cell == '^':
                splits += 1
                if c - 1 >= 0:
                    next_beams.add(c - 1)
                if c + 1 < W:
                    next_beams.add(c + 1)
            else:
                # treat any other as empty
                next_beams.add(c)
        beams = next_beams
    return splits


def part2(grid, H, W, rS, cS):
    # DP over rows: ways[c] = number of quantum timelines at position (r, c)
    if rS is None:
        return 0
    # Start just below S
    if rS + 1 >= H:
        return 1  # immediately exits
    ways = [0] * W
    ways[cS] = 1
    exit_count = 0
    for r in range(rS + 1, H):
        next_ways = [0] * W
        cell_row = grid[r]
        for c, w in enumerate(ways):
            if w == 0:
                continue
            cell = cell_row[c]
            if cell == '^':
                # split to left and right
                if c - 1 >= 0:
                    next_ways[c - 1] += w
                else:
                    exit_count += w
                if c + 1 < W:
                    next_ways[c + 1] += w
                else:
                    exit_count += w
            else:
                # continue downward
                if r + 1 < H:
                    next_ways[c] += w
                else:
                    exit_count += w
        ways = next_ways
    return exit_count


def main():
    data = read_input()
    grid, H, W, rS, cS = parse_grid(data)
    ans1 = part1(grid, H, W, rS, cS)
    ans2 = part2(grid, H, W, rS, cS)
    print(ans1)
    print(ans2)

if __name__ == '__main__':
    main()
