    # If it hits a splitter, it splits and produces beams for the *next* row?
    # Or does it hit the splitter at start_row?

    # Wait, the problem says:
    # "The incoming tachyon beam extends downward from S until it reaches the first splitter"
    # So if S is at (0, 7), and splitter is at (2, 7).
    # Row 0: S. Beam at 7.
    # Row 1: '.'. Beam at 7.
    # Row 2: '^'. Beam hits splitter. Split count++. Beams at 6, 8 created.
    # Row 3: Beams at 6, 8 interact with grid[3][6] and grid[3][8].

    # But wait, what if S is ON a splitter? (Unlikely).
    # S is just a starting point.

    # So the loop should be:
    # current_beams = {s_col}
    # for r from start_row + 1 to rows - 1:
    #    ...

    # Let's verify "S" row behavior.
    # "A tachyon beam enters the manifold at the location marked S; tachyon beams always move downward."
    # So at row `start_row`, the beam is formed. It immediately starts moving down?
    # Or does it interact with `start_row` content?
    # S is likely just 'S'. But if the input had 'S' on top of '^'?
    # The input has 'S' surrounded by '.'.
    # So we can safely start processing from start_row + 1.

    # Wait, does the beam at 'S' count as a beam?
    # "How many times will the beam be split?"
    # We only count when we hit '^'.

    current_beams = active_beams

    for r in range(start_row + 1, rows):
        next_beams = set()
        for c in current_beams:
            if c < 0 or c >= cols:
                # Beam exits manifold?
                # "until all of the tachyon beams reach a splitter or exit the manifold"
                # If c is out of bounds, it's gone.
                continue

            cell = grid[r][c]

            if cell == '^':
                split_count += 1
                next_beams.add(c - 1)
                next_beams.add(c + 1)
            else:
                # '.' or any other char (should be just empty space)
                next_beams.add(c)

        current_beams = next_beams

        if not current_beams:
            break

    print(f"Total splits: {split_count}")

    from collections import defaultdict

def solve():
    with open('input.txt', 'r') as f:
        lines = f.read().splitlines()

    grid = lines
    rows = len(grid)
    cols = len(grid[0])

    active_counts = defaultdict(int)
    start_row = 0

    found_start = False
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == 'S':
                active_counts[c] = 1
                start_row = r
                found_start = True
                break
        if found_start:
            break

    total_exited_timelines = 0

    for r in range(start_row + 1, rows):
        next_active_counts = defaultdict(int)

        # Determine all columns that have active beams
        # We only iterate over columns present in active_counts
        for c, count in active_counts.items():
            cell = grid[r][c]

            if cell == '^':
                # Split
                # Left
                left_c = c - 1
                if 0 <= left_c < cols:
                    next_active_counts[left_c] += count
                else:
                    total_exited_timelines += count

                # Right
                right_c = c + 1
                if 0 <= right_c < cols:
                    next_active_counts[right_c] += count
                else:
                    total_exited_timelines += count
            else:
                # Continue
                next_active_counts[c] += count

        active_counts = next_active_counts

    # Add remaining active beams that reached the bottom
    total_exited_timelines += sum(active_counts.values())

    print(f"Total timelines: {total_exited_timelines}")

    import math
from collections import defaultdict

def solve_and_visualize():
    with open('input.txt', 'r') as f:
        lines = f.read().splitlines()

    grid = lines
    rows = len(grid)
    cols = len(grid[0])

    active_counts = defaultdict(int)
    start_row = 0

    # Track counts for visualization: history[(r, c)] = count
    history = defaultdict(int)

    found_start = False
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == 'S':
                active_counts[c] = 1
                history[(r, c)] = 1
                start_row = r
                found_start = True
                break
        if found_start:
            break

    for r in range(start_row + 1, rows):
        next_active_counts = defaultdict(int)

        for c, count in active_counts.items():
            # Record history for visualization
            history[(r, c)] += count

            cell = grid[r][c]

            if cell == '^':
                left_c = c - 1
                if 0 <= left_c < cols:
                    next_active_counts[left_c] += count

                right_c = c + 1
                if 0 <= right_c < cols:
                    next_active_counts[right_c] += count
            else:
                next_active_counts[c] += count

        active_counts = next_active_counts

    # Generate SVG
    cell_size = 6
    width = cols * cell_size
    height = rows * cell_size

    svg_content = [f'<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}" style="background-color: #1a1a2e;">']

    max_count = max(history.values()) if history else 1
    log_max = math.log(max_count) if max_count > 1 else 1

    # Draw cells
    for (r, c), count in history.items():
        if count <= 0: continue

        # Logarithmic scaling for color
        val = math.log(count) if count > 1 else 0
        normalized = val / log_max

        # Heatmap color: Blue (low) -> Cyan -> Green -> Yellow -> Red (high)
        # Hue: 240 (blue) -> 0 (red)
        hue = 240 * (1 - normalized)

        color = f'hsl({hue}, 100%, 50%)'

        x = c * cell_size
        y = r * cell_size
        svg_content.append(f'<rect x="{x}" y="{y}" width="{cell_size}" height="{cell_size}" fill="{color}" shape-rendering="crispEdges" />')

    # Draw splitters (overlay)
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '^':
                x = c * cell_size + cell_size/2
                y = r * cell_size + cell_size/2
                # Small white dot or triangle
                svg_content.append(f'<circle cx="{x}" cy="{y}" r="{cell_size/4}" fill="white" opacity="0.5" />')
            elif grid[r][c] == 'S':
                 x = c * cell_size + cell_size/2
                 y = r * cell_size + cell_size/2
                 svg_content.append(f'<text x="{x}" y="{y + cell_size/2}" font-size="{cell_size}" fill="white" text-anchor="middle">S</text>')

    svg_content.append('</svg>')

    with open('visualization.svg', 'w') as f:
        f.write('\n'.join(svg_content))
