    #!/usr/bin/env python3
"""
Advent of Code 2025 - Day 7: Laboratories
Tachyon beam splitting simulation
"""

def solve(input_file):
    with open(input_file, 'r') as f:
        grid = [list(line.rstrip('\n')) for line in f]

    rows = len(grid)
    cols = len(grid[0]) if rows > 0 else 0

    # Find starting position S
    start_col = None
    for c in range(cols):
        if grid[0][c] == 'S':
            start_col = c
            break

    if start_col is None:
        print("Error: No starting position 'S' found!")
        return

    # Beams are tracked by their column positions
    # Each beam moves downward from its current position
    # Initially, one beam starts at column start_col, row 0

    # We'll track active beams as a set of column positions
    # When a beam hits a splitter, it produces two beams (left and right)

    # Track beams: set of column positions for current row
    active_beams = {start_col}

    split_count = 0

    # Process row by row starting from row 1
    for row in range(1, rows):
        # New beams for next row
        new_beams = set()

        for col in active_beams:
            # Check if there's a splitter at this position
            if grid[row][col] == '^':
                # Splitter: beam splits into left and right
                # Left beam goes to col-1, right beam goes to col+1
                split_count += 1
                left_col = col - 1
                right_col = col + 1
                if 0 <= left_col < cols:
                    new_beams.add(left_col)
                if 0 <= right_col < cols:
                    new_beams.add(right_col)
            else:
                # Empty space or '.': beam continues downward
                new_beams.add(col)

        active_beams = new_beams

    print(f"Total splits: {split_count}")
    return split_count

if __name__ == "__main__":
    result = solve("input.txt")

    #!/usr/bin/env python3
"""
Create visualizations for the report
"""
import math

def create_svg_beam_visualization():
    """Create SVG showing beam splitting in the example"""

    example = """.......S.......
...............
.......^.......
...............
......^.^......
...............
.....^.^.^.....
..............."""

    grid = [list(line) for line in example.strip().split('\n')]
    rows = len(grid)
    cols = len(grid)

    cell_size = 30
    width = cols * cell_size
    height = rows * cell_size

    svg = f'''<?xml version="1.0" encoding="UTF-8"?>
<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}" viewBox="0 0 {width} {height}">
  <defs>
    <linearGradient id="beamGrad" x1="0%" y1="0%" x2="0%" y2="100%">
      <stop offset="0%" style="stop-color:#00ff88;stop-opacity:1" />
      <stop offset="100%" style="stop-color:#00aa55;stop-opacity:1" />
    </linearGradient>
    <filter id="glow">
      <feGaussianBlur stdDeviation="2" result="coloredBlur"/>
      <feMerge>
        <feMergeNode in="coloredBlur"/>
        <feMergeNode in="SourceGraphic"/>
      </feMerge>
    </filter>
  </defs>

  <!-- Background -->
  <rect width="{width}" height="{height}" fill="#1a1a2e"/>

  <!-- Grid lines -->
'''

    # Grid lines
    for i in range(rows + 1):
        svg += f'  <line x1="0" y1="{i*cell_size}" x2="{width}" y2="{i*cell_size}" stroke="#333" stroke-width="1"/>\n'
    for j in range(cols + 1):
        svg += f'  <line x1="{j*cell_size}" y1="0" x2="{j*cell_size}" y2="{height}" stroke="#333" stroke-width="1"/>\n'

    # Draw splitters and start
    for row in range(rows):
        for col in range(cols):
            cell = grid[row][col]
            cx = col * cell_size + cell_size // 2
            cy = row * cell_size + cell_size // 2

            if cell == 'S':
                svg += f'  <circle cx="{cx}" cy="{cy}" r="{cell_size//3}" fill="#ffd700" filter="url(#glow)"/>\n'
                svg += f'  <text x="{cx}" y="{cy+5}" text-anchor="middle" fill="#000" font-size="14" font-weight="bold">S</text>\n'
            elif cell == '^':
                # Draw splitter as a triangle
                svg += f'  <polygon points="{cx},{cy-10} {cx-10},{cy+8} {cx+10},{cy+8}" fill="#ff6b6b" stroke="#ff9999" stroke-width="2"/>\n'

    # Draw beams simulation
    # Start position
    start_col = 7
    active_beams = {start_col}

    beam_paths = []

    for row in range(1, rows):
        new_beams = set()
        for col in active_beams:
            if grid[row][col] == '^':
                # Draw beam from previous row to splitter
                beam_paths.append((col, row-1, col, row))
                # Split
                if col - 1 >= 0:
                    new_beams.add(col - 1)
                    beam_paths.append((col, row, col-1, row))
                if col + 1 < cols:
                    new_beams.add(col + 1)
                    beam_paths.append((col, row, col+1, row))
            else:
                beam_paths.append((col, row-1, col, row))
                new_beams.add(col)
        active_beams = new_beams

    # Draw beam paths
    for (c1, r1, c2, r2) in beam_paths:
        x1 = c1 * cell_size + cell_size // 2
        y1 = r1 * cell_size + cell_size // 2
        x2 = c2 * cell_size + cell_size // 2
        y2 = r2 * cell_size + cell_size // 2
        svg += f'  <line x1="{x1}" y1="{y1}" x2="{x2}" y2="{y2}" stroke="url(#beamGrad)" stroke-width="4" stroke-linecap="round" filter="url(#glow)"/>\n'

    svg += '</svg>'

    with open('beam_visualization.svg', 'w', encoding='utf-8') as f:
        f.write(svg)

    print("Created beam_visualization.svg")


def create_timeline_tree_svg():
    """Create SVG showing how timelines multiply"""

    width = 800
    height = 500

    svg = f'''<?xml version="1.0" encoding="UTF-8"?>
<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}" viewBox="0 0 {width} {height}">
  <defs>
    <linearGradient id="treeGrad" x1="0%" y1="0%" x2="0%" y2="100%">
      <stop offset="0%" style="stop-color:#ff6b6b" />
      <stop offset="100%" style="stop-color:#ff9999" />
    </linearGradient>
    <filter id="shadow">
      <feDropShadow dx="2" dy="2" stdDeviation="3" flood-opacity="0.3"/>
    </filter>
  </defs>

  <!-- Background -->
  <rect width="{width}" height="{height}" fill="#0d1b2a"/>

  <!-- Title -->
  <text x="{width//2}" y="35" text-anchor="middle" fill="#ffd700" font-size="24" font-weight="bold">Quantum Timeline Branching</text>
  <text x="{width//2}" y="60" text-anchor="middle" fill="#aaa" font-size="14">Each splitter doubles the number of timelines</text>

'''

    # Draw tree structure
    levels = 5
    start_y = 90
    level_height = 70

    # Positions: list of (x, y) for each node
    nodes = []
    edges = []

    # Level 0: single node
    nodes.append([(width // 2, start_y)])

    for level in range(1, levels):
        prev_nodes = nodes[level - 1]
        curr_nodes = []
        spread = 300 // (2 ** (level - 1))

        for px, py in prev_nodes:
            # Left child
            lx = px - spread
            ly = py + level_height
            edges.append((px, py, lx, ly))
            curr_nodes.append((lx, ly))

            # Right child
            rx = px + spread
            ry = py + level_height
            edges.append((px, py, rx, ry))
            curr_nodes.append((rx, ry))

        nodes.append(curr_nodes)

    # Draw edges first
    for x1, y1, x2, y2 in edges:
        svg += f'  <line x1="{x1}" y1="{y1+15}" x2="{x2}" y2="{y2-15}" stroke="#4a9fff" stroke-width="2" opacity="0.7"/>\n'

    # Draw nodes
    for level_idx, level_nodes in enumerate(nodes):
        for x, y in level_nodes:
            color = "#ff6b6b" if level_idx == 0 else "#4a9fff"
            svg += f'  <circle cx="{x}" cy="{y}" r="15" fill="{color}" filter="url(#shadow)"/>\n'

    # Add labels
    svg += f'  <text x="60" y="{start_y + 5}" fill="#aaa" font-size="14">1 particle</text>\n'
    svg += f'  <text x="60" y="{start_y + level_height + 5}" fill="#aaa" font-size="14">2 timelines</text>\n'
    svg += f'  <text x="60" y="{start_y + 2*level_height + 5}" fill="#aaa" font-size="14">4 timelines</text>\n'
    svg += f'  <text x="60" y="{start_y + 3*level_height + 5}" fill="#aaa" font-size="14">8 timelines</text>\n'
    svg += f'  <text x="60" y="{start_y + 4*level_height + 5}" fill="#aaa" font-size="14">16 timelines</text>\n'

    # Formula at bottom
    svg += f'''
  <rect x="250" y="430" width="300" height="50" rx="10" fill="#1a3a5c"/>
  <text x="{width//2}" y="460" text-anchor="middle" fill="#ffd700" font-size="18">Timelines = 2^(num_splitters)</text>
'''

    svg += '</svg>'

    with open('timeline_tree.svg', 'w', encoding='utf-8') as f:
        f.write(svg)

    print("Created timeline_tree.svg")

def create_svg_beam_visualization():
    """Create SVG showing beam splitting in the example"""

    example = """.......S.......
...............
.......^.......
...............
......^.^......
...............
.....^.^.^.....
..............."""

    grid = [list(line) for line in example.strip().split('\n')]
    rows = len(grid)
    cols = len(grid)

    cell_size = 30
    width = cols * cell_size
    height = rows * cell_size

    svg = f'''<?xml version="1.0" encoding="UTF-8"?>
<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}" viewBox="0 0 {width} {height}">
  <defs>
    <linearGradient id="beamGrad" x1="0%" y1="0%" x2="0%" y2="100%">
      <stop offset="0%" style="stop-color:#00ff88;stop-opacity:1" />
      <stop offset="100%" style="stop-color:#00aa55;stop-opacity:1" />
    </linearGradient>
    <filter id="glow">
      <feGaussianBlur stdDeviation="2" result="coloredBlur"/>
      <feMerge>
        <feMergeNode in="coloredBlur"/>
        <feMergeNode in="SourceGraphic"/>
      </feMerge>
    </filter>
  </defs>

  <!-- Background -->
  <rect width="{width}" height="{height}" fill="#1a1a2e"/>

  <!-- Grid lines -->
'''

    # Grid lines
    for i in range(rows + 1):
        svg += f'  <line x1="0" y1="{i*cell_size}" x2="{width}" y2="{i*cell_size}" stroke="#333" stroke-width="1"/>\n'
    for j in range(cols + 1):
        svg += f'  <line x1="{j*cell_size}" y1="0" x2="{j*cell_size}" y2="{height}" stroke="#333" stroke-width="1"/>\n'

    # Draw splitters and start
    for row in range(rows):
        for col in range(cols):
            cell = grid[row][col]
            cx = col * cell_size + cell_size // 2
            cy = row * cell_size + cell_size // 2

            if cell == 'S':
                svg += f'  <circle cx="{cx}" cy="{cy}" r="{cell_size//3}" fill="#ffd700" filter="url(#glow)"/>\n'
                svg += f'  <text x="{cx}" y="{cy+5}" text-anchor="middle" fill="#000" font-size="14" font-weight="bold">S</text>\n'
            elif cell == '^':
                # Draw splitter as a triangle
                svg += f'  <polygon points="{cx},{cy-10} {cx-10},{cy+8} {cx+10},{cy+8}" fill="#ff6b6b" stroke="#ff9999" stroke-width="2"/>\n'

    # Draw beams simulation
    # Start position
    start_col = 7
    active_beams = {start_col}

    beam_paths = []

    for row in range(1, rows):
        new_beams = set()
        for col in active_beams:
            if grid[row][col] == '^':
                # Draw beam from previous row to splitter
                beam_paths.append((col, row-1, col, row))
                # Split
                if col - 1 >= 0:
                    new_beams.add(col - 1)
                    beam_paths.append((col, row, col-1, row))
                if col + 1 < cols:
                    new_beams.add(col + 1)
                    beam_paths.append((col, row, col+1, row))
            else:
                beam_paths.append((col, row-1, col, row))
                new_beams.add(col)
        active_beams = new_beams

    # Draw beam paths
    for (c1, r1, c2, r2) in beam_paths:
        x1 = c1 * cell_size + cell_size // 2
        y1 = r1 * cell_size + cell_size // 2
        x2 = c2 * cell_size + cell_size // 2
        y2 = r2 * cell_size + cell_size // 2
        svg += f'  <line x1="{x1}" y1="{y1}" x2="{x2}" y2="{y2}" stroke="url(#beamGrad)" stroke-width="4" stroke-linecap="round" filter="url(#glow)"/>\n'

    svg += '</svg>'

    with open('beam_visualization.svg', 'w', encoding='utf-8') as f:
        f.write(svg)

    print("Created beam_visualization.svg")


def create_timeline_tree_svg():
    """Create SVG showing how timelines multiply"""

    width = 800
    height = 500

    svg = f'''<?xml version="1.0" encoding="UTF-8"?>
<svg xmlns="http://www.w3.org/2000/svg" width="{width}" height="{height}" viewBox="0 0 {width} {height}">
  <defs>
    <linearGradient id="treeGrad" x1="0%" y1="0%" x2="0%" y2="100%">
      <stop offset="0%" style="stop-color:#ff6b6b" />
      <stop offset="100%" style="stop-color:#ff9999" />
    </linearGradient>
    <filter id="shadow">
      <feDropShadow dx="2" dy="2" stdDeviation="3" flood-opacity="0.3"/>
    </filter>
  </defs>

  <!-- Background -->
  <rect width="{width}" height="{height}" fill="#0d1b2a"/>

  <!-- Title -->
  <text x="{width//2}" y="35" text-anchor="middle" fill="#ffd700" font-size="24" font-weight="bold">Quantum Timeline Branching</text>
  <text x="{width//2}" y="60" text-anchor="middle" fill="#aaa" font-size="14">Each splitter doubles the number of timelines</text>

'''

    # Draw tree structure
    levels = 5
    start_y = 90
    level_height = 70

    # Positions: list of (x, y) for each node
    nodes = []
    edges = []

    # Level 0: single node
    nodes.append([(width // 2, start_y)])

    for level in range(1, levels):
        prev_nodes = nodes[level - 1]
        curr_nodes = []
        spread = 300 // (2 ** (level - 1))

        for px, py in prev_nodes:
            # Left child
            lx = px - spread
            ly = py + level_height
            edges.append((px, py, lx, ly))
            curr_nodes.append((lx, ly))

            # Right child
            rx = px + spread
            ry = py + level_height
            edges.append((px, py, rx, ry))
            curr_nodes.append((rx, ry))

        nodes.append(curr_nodes)

    # Draw edges first
    for x1, y1, x2, y2 in edges:
        svg += f'  <line x1="{x1}" y1="{y1+15}" x2="{x2}" y2="{y2-15}" stroke="#4a9fff" stroke-width="2" opacity="0.7"/>\n'

    # Draw nodes
    for level_idx, level_nodes in enumerate(nodes):
        for x, y in level_nodes:
            color = "#ff6b6b" if level_idx == 0 else "#4a9fff"
            svg += f'  <circle cx="{x}" cy="{y}" r="15" fill="{color}" filter="url(#shadow)"/>\n'

    # Add labels
    svg += f'  <text x="60" y="{start_y + 5}" fill="#aaa" font-size="14">1 particle</text>\n'
    svg += f'  <text x="60" y="{start_y + level_height + 5}" fill="#aaa" font-size="14">2 timelines</text>\n'
    svg += f'  <text x="60" y="{start_y + 2*level_height + 5}" fill="#aaa" font-size="14">4 timelines</text>\n'
    svg += f'  <text x="60" y="{start_y + 3*level_height + 5}" fill="#aaa" font-size="14">8 timelines</text>\n'
    svg += f'  <text x="60" y="{start_y + 4*level_height + 5}" fill="#aaa" font-size="14">16 timelines</text>\n'

    # Formula at bottom
    svg += f'''
  <rect x="250" y="430" width="300" height="50" rx="10" fill="#1a3a5c"/>
  <text x="{width//2}" y="460" text-anchor="middle" fill="#ffd700" font-size="18">Timelines = 2^(num_splitters)</text>
'''

    svg += '</svg>'

    with open('timeline_tree.svg', 'w', encoding='utf-8') as f:
        f.write(svg)

    print("Created timeline_tree.svg")
