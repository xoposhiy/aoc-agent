    Summary

    The task involves selecting a specific number of batteries from a bank (represented as a string of digits) to form the largest possible number. The constraint is that the relative order of the selected digits must be preserved.

    • Part 1: Select exactly 2 digits.
    • Part 2: Select exactly 12 digits.

    The goal is to maximize the numeric value of the concatenation of the selected digits, summed across all banks.

    Algorithmic Approach

    The problem asks for the lexicographically largest subsequence of length \(K\). This is a classic problem that can be solved with a Greedy Algorithm.

    Strategy

    To form the largest number, we want the most significant digit (leftmost) to be as large as possible. However, picking a digit at index \(i\) restricts us to picking subsequent digits only from indices \(> i\). We must ensure that if we pick a digit at index \(i\), there are enough remaining characters (at least \(K-1\)) in the string after \(i\) to complete the sequence.

    Algorithm

    1. Let \(K\) be the number of digits needed.
    2. Maintain a current_position in the string (initially 0).
    3. Repeat \(K\) times:
      • Determine the valid search window: from current_position to len(string) - needed.
        • The end of the window is determined by the requirement to leave needed - 1 characters for the future.
      • Find the largest digit in this window.
      • If there are multiple occurrences of the largest digit, pick the first one (to leave the maximum number of options for subsequent steps, although any valid occurrence works in this greedy setup).
      • Append this digit to our result.
      • Update current_position to index_of_chosen_digit + 1.
      • Decrement needed.

    This approach runs in \(O(K \cdot N)\) time per line, where \(N\) is the string length. Since \(K\) is small (2 or 12) and \(N\) is small, this is extremely efficient.

    Key Insights

    • Greedy Choice Property: At any step, choosing the largest possible valid digit never prevents us from finding an optimal solution for the remaining suffix. In fact, maximizing the current digit is strictly better than "saving" a larger digit for a less significant position.
    • Window constraint: The "gotcha" is ensuring you don't pick a digit so late in the string that you run out of characters to fill the remaining slots. The search window [current_pos, len - needed] handles this perfectly.

def solve_greedy(digits, k):
    needed = k
    current_pos = 0
    indices = []

    while needed > 0:
        search_end = len(digits) - needed
        window = digits[current_pos : search_end + 1]
        best_digit = max(window)
        best_rel_index = window.index(best_digit)

        real_index = current_pos + best_rel_index
        indices.append(real_index)

        current_pos = real_index + 1
        needed -= 1
    return indices

    def solve():
    try:
        with open('input.txt', 'r') as f:
            lines = f.read().splitlines()
    except FileNotFoundError:
        print("Input file not found.")
        return

    total_joltage = 0

    for line in lines:
        if not line:
            continue

        # Parse digits
        digits = [int(c) for c in line if c.isdigit()]

        if len(digits) < 2:
            continue

        max_val = -1

        # We iterate through possible first digits from 9 down to 1
        # To optimize, we find the *first* occurrence of each digit.
        # Choosing the first occurrence leaves the maximum remaining string for the second digit.

        for d1 in range(9, 0, -1):
            try:
                # Find first index of d1
                idx = digits.index(d1)
            except ValueError:
                continue

            # Look for the max digit in the rest of the list
            remaining = digits[idx+1:]
            if not remaining:
                continue

            d2 = max(remaining)
            val = d1 * 10 + d2
            if val > max_val:
                max_val = val

            # Pruning:
            # The maximum possible value we can find with any start digit <= d1-1
            # is (d1-1)*10 + 9.
            # If our current max_val is greater than that, we don't need to check further.
            if max_val >= (d1 - 1) * 10 + 9:
                break

        total_joltage += max_val

    print(f"Part 1: {total_joltage}")

if __name__ == '__main__':

    def solve():
    try:
        with open('input.txt', 'r') as f:
            lines = f.read().splitlines()
    except FileNotFoundError:
        print("Input file not found.")
        return

    total_joltage = 0

    for line_idx, line in enumerate(lines):
        if not line:
            continue

        # Parse digits
        # Although problem says 1-9, input might contain anything? 
        # "labeled with their joltage rating... digits in your input"
        # I'll stick to extracting digits.
        digits = [int(c) for c in line if c.isdigit()]

        if len(digits) < 12:
            # Cannot select 12 batteries
            continue

        # We need to pick 12 digits.
        # Greedy approach.

        current_digits = digits
        result_digits = []
        needed = 12

        # Start scanning from index 0
        current_pos = 0

        while needed > 0:
            # We must pick a digit from current_pos up to an index
            # that leaves enough digits for the remaining 'needed - 1'
            # Remaining length after picking index i is: len(digits) - 1 - i
            # We need: len(digits) - 1 - i >= needed - 1
            # => len(digits) - i >= needed
            # => i <= len(digits) - needed

            # So search range is [current_pos, len(digits) - needed] inclusive

            search_end = len(digits) - needed

            # Find max digit in digits[current_pos : search_end + 1]
            # We want the *first* occurrence of the max digit to maximize opportunities?
            # Actually, as discussed, first occurrence is optimal.

            window = digits[current_pos : search_end + 1]

            # Find max value in window
            best_digit = -1
            best_rel_index = -1

            # Since window is list of ints, max() works
            if not window:
                # Should not happen if logic is correct
                break

            best_digit = max(window)

            # Find first index of best_digit in window
            best_rel_index = window.index(best_digit)

            # Add to result
            result_digits.append(best_digit)

            # Update current_pos
            # The actual index in 'digits' is current_pos + best_rel_index
            # We continue from the next position
            current_pos = current_pos + best_rel_index + 1

            needed -= 1

        # Convert result_digits to number
        # They are digits 0-9
        val_str = "".join(str(d) for d in result_digits)
        val = int(val_str)

        total_joltage += val

    print(f"Part 2: {total_joltage}")
