# Solution in Python
#
# Approach Part 1:
# For each line (bank) consisting of digits '1'..'9', we must select two digits in order (i<j)
# to form the largest possible two-digit number 10*a + b. We cannot rearrange digits.
# The optimal strategy is greedy: for each possible left digit a (from left to right),
# pair it with the maximum possible right digit b appearing after it. To compute this
# efficiently, scan the string from right to left maintaining best_right[d] = whether there
# exists a digit >= d to the right, or more directly maintain max_right_digit_at_pos[i] which
# is the maximum digit on suffix starting at i. Then, for each position i as left digit a,
# best pair is with max_right_digit_on_suffix(i+1). Keep the maximum value over i. This is
# O(n) per line.
# Edge cases: Every line length is >= 2 per problem context. Digits are 1..9.
#
# Approach Part 2: (will update after reading statement). For now, we structure code to reuse parsing and helpers.
#
# The program reads input.txt, computes Part 1, prints results.
from __future__ import annotations
import sys
from typing import List


def parse_input(text: str) -> List[str]:
    return [line.strip() for line in text.strip().splitlines() if line.strip()]


def max_two_digit_from_sequence(s: str) -> int:
    # digits are '1'..'9'
    n = len(s)
    # suffix max digit values (as int) starting at position i
    # suf_max[i] = max digit value in s[i:]
    suf_max = [0]*(n+1)
    for i in range(n-1, -1, -1):
        suf_max[i] = max(suf_max[i+1], ord(s[i]) - 48)
    best = -1
    left_best_digit_needed = None  # not needed for answer, could store pair if desired
    for i in range(n-1):
        a = ord(s[i]) - 48
        b = suf_max[i+1]
        val = a*10 + b
        if val > best:
            best = val
            left_best_digit_needed = a
            # if best reaches 99, we can early return
            if best == 99:
                return best
    return best


def solve_part1(lines: List[str]) -> int:
    return sum(max_two_digit_from_sequence(s) for s in lines)

        b = suf_max[i+1]
        val = a*10 + b
        if val > best:
            best = val
            if best == 99:
                return best
    return best


def solve_part1(lines: List[str]) -> int:
    return sum(max_two_digit_from_sequence(s) for s in lines)


def best_k_subsequence_number(s: str, k: int) -> int:
    n = len(s)
    assert n >= k, f"Line shorter than {k}: {s}"
    rem = n - k
    stack: List[str] = []
    for ch in s:
        while rem and stack and stack[-1] < ch:
            stack.pop()
            rem -= 1
        stack.append(ch)
    # if still need to remove, remove from end (smallest impact on lexicographic order)
    if rem:
        stack = stack[:-rem]
    # compose first k digits
    res = 0
    for i in range(k):
        res = res*10 + (ord(stack[i]) - 48)
    return res


def solve_part2(lines: List[str]) -> int:
    K = 12
    total = 0
    for s in lines:
        total += best_k_subsequence_number(s, K)
    return total

def best_k_subsequence_steps(s: str, k: int) -> Tuple[List[List[str]], List[int]]:
    # returns snapshots of the stack and remaining removals after each char
    n = len(s)
    rem = n - k
    stack: List[str] = []
    snapshots: List[List[str]] = []
    rem_left: List[int] = []
    for ch in s:
        while rem and stack and stack[-1] < ch:
            stack.pop()
            rem -= 1
        stack.append(ch)
        snapshots.append(stack.copy())
        rem_left.append(rem)
    if rem:
        # simulate trimming
        stack = stack[:-rem]
        snapshots.append(stack.copy())
        rem_left.append(0)
    return snapshots, rem_left

def monotone_pick_mask(s: str, k: int) -> List[bool]:
    n = len(s)
    rem = n - k
    stack_idx: List[int] = []
    for i, ch in enumerate(s):
        while rem and stack_idx and s[stack_idx[-1]] < ch:
            stack_idx.pop()
            rem -= 1
        stack_idx.append(i)
    if rem:
        stack_idx = stack_idx[:-rem]
    keep = [False]*n
    for i in stack_idx[:k]:
        keep[i] = True
    return keep
