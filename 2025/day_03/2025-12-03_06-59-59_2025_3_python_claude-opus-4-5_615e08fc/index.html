    """
Create SVG visualization for Day 3: Lobby (Battery Selection)
"""

def create_battery_svg():
    """Create an educational SVG showing the greedy algorithm for selecting digits."""

    # Example line
    line = "818181911112111"
    k = 12

    # Run algorithm to get selection
    n = len(line)
    selected = []
    start = 0

    for remaining in range(k, 0, -1):
        end = n - remaining
        best_pos = start
        best_digit = line[start]
        for pos in range(start, end + 1):
            if line[pos] > best_digit:
                best_digit = line[pos]
                best_pos = pos
        selected.append(best_pos)
        start = best_pos + 1

    selected_set = set(selected)
    result = ''.join(line[i] for i in selected)

    # SVG dimensions
    cell_w = 40
    cell_h = 50
    margin = 20
    width = margin * 2 + len(line) * cell_w + 50
    height = 350

    svg_parts = [
        f'<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 {width} {height}">',
        '<style>',
        '  .title { font: bold 18px sans-serif; fill: #2c3e50; }',
        '  .subtitle { font: 14px sans-serif; fill: #7f8c8d; }',
        '  .digit { font: bold 20px monospace; }',
        '  .selected { fill: #27ae60; }',
        '  .unselected { fill: #e74c3c; }',
        '  .battery-sel { fill: #2ecc71; stroke: #27ae60; stroke-width: 2; }',
        '  .battery-unsel { fill: #ffcccc; stroke: #e74c3c; stroke-width: 2; }',
        '  .result { font: bold 16px monospace; fill: #2c3e50; }',
        '  .label { font: 12px sans-serif; fill: #34495e; }',
        '  .arrow { stroke: #3498db; stroke-width: 2; fill: none; marker-end: url(#arrowhead); }',
        '</style>',
        '<defs>',
        '  <marker id="arrowhead" markerWidth="10" markerHeight="7" refX="9" refY="3.5" orient="auto">',
        '    <polygon points="0 0, 10 3.5, 0 7" fill="#3498db"/>',
        '  </marker>',
        '</defs>',

        # Title
        f'<text x="{width/2}" y="30" text-anchor="middle" class="title">Greedy Battery Selection Algorithm</text>',
        f'<text x="{width/2}" y="50" text-anchor="middle" class="subtitle">Select 12 digits to form maximum number</text>',

        # Original string label
        '<text x="20" y="90" class="label">Input bank:</text>',
    ]

    # Draw batteries
    y_base = 100
    for i, d in enumerate(line):
        x = margin + i * cell_w
        cls = "battery-sel" if i in selected_set else "battery-unsel"

        # Battery body
        svg_parts.append(f'<rect x="{x+5}" y="{y_base}" width="{cell_w-10}" height="{cell_h}" rx="5" class="{cls}"/>')
        # Battery top (terminal)
        svg_parts.append(f'<rect x="{x+12}" y="{y_base-8}" width="{cell_w-24}" height="10" rx="2" fill="#95a5a6"/>')

        # Digit
        txt_cls = "selected" if i in selected_set else "unselected"
        svg_parts.append(f'<text x="{x + cell_w/2}" y="{y_base + cell_h/2 + 7}" text-anchor="middle" class="digit {txt_cls}">{d}</text>')

        # Position index
        svg_parts.append(f'<text x="{x + cell_w/2}" y="{y_base + cell_h + 15}" text-anchor="middle" class="label">{i}</text>')

    # Legend
    y_legend = 195
    svg_parts.append(f'<rect x="20" y="{y_legend}" width="20" height="20" rx="3" class="battery-sel"/>')
    svg_parts.append(f'<text x="50" y="{y_legend + 15}" class="label">Selected (12 digits)</text>')
    svg_parts.append(f'<rect x="180" y="{y_legend}" width="20" height="20" rx="3" class="battery-unsel"/>')
    svg_parts.append(f'<text x="210" y="{y_legend + 15}" class="label">Not selected</text>')

    # Result
    y_result = 250
    svg_parts.append(f'<text x="20" y="{y_result}" class="label">Result: </text>')
    svg_parts.append(f'<text x="70" y="{y_result}" class="result">{result}</text>')

    # Algorithm steps explanation
    y_algo = 285
    svg_parts.append(f'<text x="20" y="{y_algo}" class="subtitle">Algorithm: At each step, pick the largest digit</text>')
    svg_parts.append(f'<text x="20" y="{y_algo + 20}" class="subtitle">that leaves enough digits remaining to complete the selection.</text>')

    # Show the key insight
    y_insight = 335
    svg_parts.append(f'<text x="{width/2}" y="{y_insight}" text-anchor="middle" class="label"> Max joltage from "818181911112111": 888911112111 </text>')

    svg_parts.append('</svg>')

    return '\n'.join(svg_parts)

Day 3, Part 1: Find maximum 2-digit number from each bank by selecting exactly 2 batteries.

Approach:
- For each line, find the maximum digit that has at least one digit after it (tens digit)
- Then find the maximum digit after that position (units digit)
- Sum up all the 2-digit numbers
"""

def max_joltage(line):
    """Find maximum 2-digit number by selecting 2 digits in order."""
    n = len(line)
    best = 0
    # Try each position for the first digit (tens)
    for i in range(n - 1):
        tens = int(line[i])
        # Find max digit after position i
        units = max(int(line[j]) for j in range(i + 1, n))
        joltage = tens * 10 + units
        best = max(best, joltage)
    return best

# Read input
with open("input.txt") as f:
    lines = [line.strip() for line in f if line.strip()]

# Calculate total
total = sum(max_joltage(line) for line in lines)
print(f"Part 1: {total}")

    """
Day 3, Part 2: Find maximum 12-digit number from each bank by selecting exactly 12 batteries.

Approach (greedy):
- We need to pick 12 digits in order from left to right
- At each step, pick the maximum digit that still allows selecting 12 digits total
- If we need to pick (12-k) more digits starting from position 'start',
  we can choose from positions [start, n-(12-k)] to ensure enough digits remain
"""

def max_joltage_k(line, k=12):
    """Find maximum k-digit number by selecting k digits in order (greedy)."""
    n = len(line)
    result = []
    start = 0

    for remaining in range(k, 0, -1):
        # We need 'remaining' more digits
        # Can pick from start to n-remaining (inclusive)
        end = n - remaining

        # Find max digit in range [start, end]
        best_pos = start
        best_digit = line[start]
        for pos in range(start, end + 1):
            if line[pos] > best_digit:
                best_digit = line[pos]
                best_pos = pos

        result.append(best_digit)
        start = best_pos + 1  # Next digit must be after this position

    return int(''.join(result))
