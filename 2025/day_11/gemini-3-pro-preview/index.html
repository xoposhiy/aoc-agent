    import sys
from collections import defaultdict

sys.setrecursionlimit(20000)

def solve():
    adj = defaultdict(list)
    try:
        with open('input.txt', 'r') as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                parts = line.split(':')
                src = parts[0].strip()
                dests = parts[1].strip().split()
                for d in dests:
                    adj[src].append(d)
    except FileNotFoundError:
        print("input.txt not found")
        return

    memo = {}
    visiting = set()

    # Check for cycles and count paths
    def count_paths(u):
        if u == 'out':
            return 1
        if u in memo:
            return memo[u]
        if u in visiting:
            # Cycle detected
            print(f"Cycle detected involving {u}")
            return None # Or raise error

        visiting.add(u)
        total_paths = 0
        if u in adj:
            for v in adj[u]:
                res = count_paths(v)
                if res is None:
                    return None
                total_paths += res

        visiting.remove(u)
        memo[u] = total_paths
        return total_paths

    result = count_paths('you')
    print(f"Total paths from 'you' to 'out': {result}")

if __name__ == '__main__':
    solve()

    import matplotlib.pyplot as plt
import networkx as nx
import random

def create_viz():
    G = nx.DiGraph()

    # Define main nodes
    nodes = ['svr', 'fft', 'dac', 'out']
    pos = {
        'svr': (0, 10),
        'fft': (0, 6),
        'dac': (0, 2),
        'out': (0, -2)
    }

    # Add main path edges
    G.add_edge('svr', 'fft', weight=2)
    G.add_edge('fft', 'dac', weight=2)
    G.add_edge('dac', 'out', weight=2)

    # Add some "cloud" nodes to represent complexity
    for i in range(5):
        node_name = f'n{i}'
        # Scatter them around the first segment
        G.add_node(node_name)
        pos[node_name] = (random.uniform(-3, 3), random.uniform(7, 9))
        if random.random() > 0.5:
            G.add_edge('svr', node_name)
            G.add_edge(node_name, 'fft')
        else:
            G.add_edge('svr', 'fft')

    for i in range(5, 10):
        node_name = f'n{i}'
        # Scatter them around the second segment
        G.add_node(node_name)
        pos[node_name] = (random.uniform(-3, 3), random.uniform(3, 5))
        if random.random() > 0.5:
            G.add_edge('fft', node_name)
            G.add_edge(node_name, 'dac')
        else:
            G.add_edge('fft', 'dac')

    for i in range(10, 15):
        node_name = f'n{i}'
        # Scatter them around the third segment
        G.add_node(node_name)
        pos[node_name] = (random.uniform(-3, 3), random.uniform(-1, 1))
        if random.random() > 0.5:
            G.add_edge('dac', node_name)
            G.add_edge(node_name, 'out')
        else:
            G.add_edge('dac', 'out')

    plt.figure(figsize=(8, 12))

    # Draw nodes
    nx.draw_networkx_nodes(G, pos, nodelist=nodes, node_color='#ff6b6b', node_size=3000, alpha=1.0)
    nx.draw_networkx_nodes(G, pos, nodelist=[n for n in G.nodes if n not in nodes], node_color='#4ecdc4', node_size=500, alpha=0.6)

    # Draw labels
    nx.draw_networkx_labels(G, pos, labels={n:n for n in nodes}, font_size=16, font_weight='bold', font_color='white')

    # Draw edges
    nx.draw_networkx_edges(G, pos, edge_color='#556270', arrows=True, arrowsize=20, width=1.5, alpha=0.7)

    # Annotate counts
    plt.text(2.5, 8, f"Paths(svr->fft)\n= 21,788", fontsize=12, bbox=dict(facecolor='white', alpha=0.8))
    plt.text(2.5, 4, f"Paths(fft->dac)\n= 6,402,587", fontsize=12, bbox=dict(facecolor='white', alpha=0.8))
    plt.text(2.5, 0, f"Paths(dac->out)\n= 3,651", fontsize=12, bbox=dict(facecolor='white', alpha=0.8))

    plt.text(0, -3.5, "Total = A * B * C\n= 509,312,913,844,956", ha='center', fontsize=14, fontweight='bold', bbox=dict(facecolor='#ffe66d', alpha=1.0))

    plt.title("Day 11: The Flow of Data", fontsize=20)
    plt.axis('off')
    plt.tight_layout()
    plt.savefig('graph_solution.png')

if __name__ == '__main__':
    create_viz()
